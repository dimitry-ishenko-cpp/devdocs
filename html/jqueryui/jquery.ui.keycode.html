Example:

Test for arrow key presses.

$( document ).on( "keydown", function( event ) {
  switch( event.keyCode ) {
    case $.ui.keyCode.LEFT:
      console.log( "left" );
      break;
    case $.ui.keyCode.RIGHT:
      console.log( "right" );
      break;
    case $.ui.keyCode.UP:
      console.log( "up" );
      break;
    case $.ui.keyCode.DOWN:
      console.log( "down" );
      break;
  }
});
