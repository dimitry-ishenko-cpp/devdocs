Initialize $wp_scripts if it has not been set.

Return

WP_Scripts WP_Scripts instance.

File: wp-includes/functions.wp-scripts.php. View all references 

function wp_scripts() {
	global $wp_scripts;

	if ( ! ( $wp_scripts instanceof WP_Scripts ) ) {
		$wp_scripts = new WP_Scripts();
	}

	return $wp_scripts;
}
