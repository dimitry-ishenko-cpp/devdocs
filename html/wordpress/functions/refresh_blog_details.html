Clear the blog details cache.

Parameters

$blog_id int Optional
Blog ID. Defaults to current blog.

function refresh_blog_details( $blog_id = 0 ) {
	$blog_id = (int) $blog_id;
	if ( ! $blog_id ) {
		$blog_id = get_current_blog_id();
	}

	clean_blog_cache( $blog_id );
}
