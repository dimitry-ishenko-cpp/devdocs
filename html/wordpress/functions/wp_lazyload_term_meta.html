Parameters

function wp_lazyload_term_meta( array $term_ids ) {
	if ( empty( $term_ids ) ) {
		return;
	}
	$lazyloader = wp_metadata_lazyloader();
	$lazyloader->queue_objects( 'term', $term_ids );
}
