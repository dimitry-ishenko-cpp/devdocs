wp_update_network_site_counts( int|null $network_id = null )

Updates the network-wide site count.

Parameters

$network_id int|null Optional
ID of the network. Default is the current network.

Default: null

Source

File: wp-includes/ms-functions.php. View all references 

function wp_update_network_site_counts( $network_id = null ) {
	$network_id = (int) $network_id;
	if ( ! $network_id ) {
		$network_id = get_current_network_id();
	}

	$count = get_sites(
		array(
			'network_id'             => $network_id,
			'spam'                   => 0,
			'deleted'                => 0,
			'archived'               => 0,
			'count'                  => true,
			'update_site_meta_cache' => false,
		)
	);

	update_network_option( $network_id, 'blog_count', $count );
}
