_wp_http_get_object(): WP_Http

Returns the initialized WP_Http Object

Return

WP_Http HTTP Transport object.

Source

function _wp_http_get_object() {
	static $http = null;

	if ( is_null( $http ) ) {
		$http = new WP_Http();
	}
	return $http;
}
