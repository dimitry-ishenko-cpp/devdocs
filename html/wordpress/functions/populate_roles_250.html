populate_roles_250()

Create and modify WordPress roles for WordPress 2.5.

Source

function populate_roles_250() {
	$role = get_role( 'administrator' );

	if ( ! empty( $role ) ) {
		$role->add_cap( 'edit_dashboard' );
	}
}
