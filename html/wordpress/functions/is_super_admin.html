is_super_admin( int|false $user_id = false ): bool

Determines whether user is a site admin.

Parameters

$user_id int|false Optional
The ID of a user. Defaults to false, to check the current user.

Default: false

Return

bool Whether the user is a site admin.

Source

File: wp-includes/capabilities.php. View all references 

function is_super_admin( $user_id = false ) {
	if ( ! $user_id ) {
		$user = wp_get_current_user();
	} else {
		$user = get_userdata( $user_id );
	}

	if ( ! $user || ! $user->exists() ) {
		return false;
	}

	if ( is_multisite() ) {
		$super_admins = get_super_admins();
		if ( is_array( $super_admins ) && in_array( $user->user_login, $super_admins, true ) ) {
			return true;
		}
	} else {
		if ( $user->has_cap( 'delete_users' ) ) {
			return true;
		}
	}

	return false;
}
