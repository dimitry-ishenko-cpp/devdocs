populate_roles_280()

Create and modify WordPress roles for WordPress 2.8.

Source

function populate_roles_280() {
	$role = get_role( 'administrator' );

	if ( ! empty( $role ) ) {
		$role->add_cap( 'install_themes' );
	}
}
