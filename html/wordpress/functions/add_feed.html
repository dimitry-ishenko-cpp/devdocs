add_feed( string $feedname, callable $function ): string

Adds a new feed type like /atom1/.

Parameters

$feedname string Required
Feed name.
$function callable Required
Callback to run on feed display.

Return

string Feed action name.

More Information

Requires one-time use of flush_rules() to take effect.

Source

File: wp-includes/rewrite.php. View all references 

function add_feed( $feedname, $function ) {
	global $wp_rewrite;

	if ( ! in_array( $feedname, $wp_rewrite->feeds, true ) ) {
		$wp_rewrite->feeds[] = $feedname;
	}

	$hook = 'do_feed_' . $feedname;

	// Remove default function hook.
	remove_action( $hook, $hook );

	add_action( $hook, $function, 10, 2 );

	return $hook;
}
