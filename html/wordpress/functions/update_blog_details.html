update_blog_details( int $blog_id, array $details = array() ): bool

Updates the details for a blog and the blogs table for a given blog ID.

Parameters

$blog_idintrequired
Blog ID.
$detailsarrayoptional
Array of details keyed by blogs table field names.

Default:array()

Return

Source

function update_blog_details( $blog_id, $details = array() ) {
	if ( empty( $details ) ) {
		return false;
	}

	if ( is_object( $details ) ) {
		$details = get_object_vars( $details );
	}

	$site = wp_update_site( $blog_id, $details );

	if ( is_wp_error( $site ) ) {
		return false;
	}

	return true;
}
