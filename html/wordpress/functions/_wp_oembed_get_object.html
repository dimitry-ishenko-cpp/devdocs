_wp_oembed_get_object(): WP_oEmbed

Returns the initialized WP_oEmbed object.

Return

WP_oEmbed object.

Source

function _wp_oembed_get_object() {
	static $wp_oembed = null;

	if ( is_null( $wp_oembed ) ) {
		$wp_oembed = new WP_oEmbed();
	}
	return $wp_oembed;
}
