wxr_cdata( string $str ): string

Wraps given string in XML CDATA tag.

Parameters

$str string Required
String to wrap in XML CDATA tag.

Return

string

Source

function wxr_cdata( $str ) {
	if ( ! seems_utf8( $str ) ) {
		$str = utf8_encode( $str );
	}
	// $str = ent2ncr(esc_html($str));
	$str = '<![CDATA[' . str_replace( ']]>', ']]]]><![CDATA[>', $str ) . ']]>';

	return $str;
}
