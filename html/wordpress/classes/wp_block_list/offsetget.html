public function offsetGet( $offset ) {
	$block = $this->blocks[ $offset ];

	if ( isset( $block ) && is_array( $block ) ) {
		$block = new WP_Block( $block, $this->available_context, $this->registry );

		$this->blocks[ $offset ] = $block;
	}

	return $block;
}
