public function get_by_slug( $template_slug ) {
	$all_templates = $this->get_all_registered();

	if ( ! $all_templates ) {
		return null;
	}

	foreach ( $all_templates as $template ) {
		if ( $template->slug === $template_slug ) {
			return $template;
		}
	}

	return null;
}
