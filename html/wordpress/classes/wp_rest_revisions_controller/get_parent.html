WP_REST_Revisions_Controller::get_parent( int $parent ): WP_Post|WP_Error

Get the parent post, if the ID is valid.

Parameters

$parent int Required
Supplied ID.

Return

WP_Post|WP_Error Post object if ID is valid, WP_Error otherwise.

Source

protected function get_parent( $parent ) {
	$error = new WP_Error(
		'rest_post_invalid_parent',
		__( 'Invalid post parent ID.' ),
		array( 'status' => 404 )
	);
	if ( (int) $parent <= 0 ) {
		return $error;
	}

	$parent = get_post( (int) $parent );
	if ( empty( $parent ) || empty( $parent->ID ) || $this->parent_post_type !== $parent->post_type ) {
		return $error;
	}

	return $parent;
}
