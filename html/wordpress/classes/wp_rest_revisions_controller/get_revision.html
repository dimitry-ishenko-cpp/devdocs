protected function get_revision( $id ) {
	$error = new WP_Error(
		'rest_post_invalid_id',
		__( 'Invalid revision ID.' ),
		array( 'status' => 404 )
	);

	if ( (int) $id <= 0 ) {
		return $error;
	}

	$revision = get_post( (int) $id );
	if ( empty( $revision ) || empty( $revision->ID ) || 'revision' !== $revision->post_type ) {
		return $error;
	}

	return $revision;
}
