public function export_original( $entry ) {
	// TODO: Warnings for control characters.
	$exported = $entry->singular;
	if ( $entry->is_plural ) {
		$exported .= "\0" . $entry->plural;
	}
	if ( $entry->context ) {
		$exported = $entry->context . "\4" . $exported;
	}
	return $exported;
}
