public function __construct( $args = array() ) {
	$defaults = array(
		'type'      => 'web',
		'url'       => '',
		'plugin'    => '',
		'nonce'     => '',
		'title'     => '',
		'overwrite' => '',
	);
	$args     = wp_parse_args( $args, $defaults );

	$this->type      = $args['type'];
	$this->url       = $args['url'];
	$this->api       = isset( $args['api'] ) ? $args['api'] : array();
	$this->overwrite = $args['overwrite'];

	parent::__construct( $args );
}
