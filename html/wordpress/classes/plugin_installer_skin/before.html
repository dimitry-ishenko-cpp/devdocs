Plugin_Installer_Skin::before()

Performs an action before installing a plugin.

Source

public function before() {
	if ( ! empty( $this->api ) ) {
		$this->upgrader->strings['process_success'] = sprintf(
			$this->upgrader->strings['process_success_specific'],
			$this->api->name,
			$this->api->version
		);
	}
}
