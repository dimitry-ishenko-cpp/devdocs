protected static function get_empty_value_for_type( $type ) {
	switch ( $type ) {
		case 'string':
			return '';
		case 'boolean':
			return false;
		case 'integer':
			return 0;
		case 'number':
			return 0.0;
		case 'array':
		case 'object':
			return array();
		default:
			return null;
	}
}
