public function register_field() {
	_deprecated_function( __METHOD__, '5.6.0' );

	register_rest_field(
		$this->get_rest_field_type(),
		'meta',
		array(
			'get_callback'    => array( $this, 'get_value' ),
			'update_callback' => array( $this, 'update_value' ),
			'schema'          => $this->get_field_schema(),
		)
	);
}
