protected function coordinates_match( $a, $b ) {
	if ( ! isset( $a['latitude'], $a['longitude'], $b['latitude'], $b['longitude'] ) ) {
		return false;
	}

	return $a['latitude'] === $b['latitude'] && $a['longitude'] === $b['longitude'];
}
