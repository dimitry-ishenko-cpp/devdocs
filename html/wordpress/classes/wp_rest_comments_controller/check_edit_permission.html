protected function check_edit_permission( $comment ) {
	if ( 0 === (int) get_current_user_id() ) {
		return false;
	}

	if ( current_user_can( 'moderate_comments' ) ) {
		return true;
	}

	return current_user_can( 'edit_comment', $comment->comment_ID );
}
