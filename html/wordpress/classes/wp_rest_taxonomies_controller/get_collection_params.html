public function get_collection_params() {
	$new_params            = array();
	$new_params['context'] = $this->get_context_param( array( 'default' => 'view' ) );
	$new_params['type']    = array(
		'description' => __( 'Limit results to taxonomies associated with a specific post type.' ),
		'type'        => 'string',
	);
	return $new_params;
}
