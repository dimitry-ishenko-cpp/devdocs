public function get_item( $request ) {
	$tax_obj = get_taxonomy( $request['taxonomy'] );

	if ( empty( $tax_obj ) ) {
		return new WP_Error(
			'rest_taxonomy_invalid',
			__( 'Invalid taxonomy.' ),
			array( 'status' => 404 )
		);
	}

	$data = $this->prepare_item_for_response( $tax_obj, $request );

	return rest_ensure_response( $data );
}
