protected function prepare_links( $taxonomy ) {
	return array(
		'collection'              => array(
			'href' => rest_url( sprintf( '%s/%s', $this->namespace, $this->rest_base ) ),
		),
		'https://api.w.org/items' => array(
			'href' => rest_url( rest_get_route_for_taxonomy_items( $taxonomy->name ) ),
		),
	);
}
