public function json() {
	$exports = array(
		'settings'           => $this->settings,
		'primarySetting'     => $this->primary_setting,
		'selector'           => $this->selector,
		'type'               => $this->type,
		'fallbackRefresh'    => $this->fallback_refresh,
		'containerInclusive' => $this->container_inclusive,
	);
	return $exports;
}
