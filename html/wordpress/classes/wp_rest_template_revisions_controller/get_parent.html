protected function get_parent( $parent_template_id ) {
	$template = get_block_template( $parent_template_id, $this->parent_post_type );

	if ( ! $template ) {
		return new WP_Error(
			'rest_post_invalid_parent',
			__( 'Invalid template parent ID.' ),
			array( 'status' => 404 )
		);
	}

	return get_post( $template->wp_id );
}
