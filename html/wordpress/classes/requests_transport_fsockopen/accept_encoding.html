protected static function accept_encoding() {
	$type = array();
	if (function_exists('gzinflate')) {
		$type[] = 'deflate;q=1.0';
	}

	if (function_exists('gzuncompress')) {
		$type[] = 'compress;q=0.5';
	}

	$type[] = 'gzip;q=0.5';

	return implode(', ', $type);
}
