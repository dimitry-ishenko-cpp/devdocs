protected function _column_title( $post, $classes, $data, $primary ) {
	echo '<td class="' . $classes . ' page-title" ', $data, '>';
	echo $this->column_title( $post );
	echo $this->handle_row_actions( $post, 'title', $primary );
	echo '</td>';
}
