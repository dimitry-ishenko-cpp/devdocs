public function validate_plugin_param( $file ) {
	if ( ! is_string( $file ) || ! preg_match( '/' . self::PATTERN . '/u', $file ) ) {
		return false;
	}

	$validated = validate_file( plugin_basename( $file ) );

	return 0 === $validated;
}
