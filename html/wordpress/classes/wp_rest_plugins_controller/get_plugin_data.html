protected function get_plugin_data( $plugin ) {
	$plugins = get_plugins();

	if ( ! isset( $plugins[ $plugin ] ) ) {
		return new WP_Error( 'rest_plugin_not_found', __( 'Plugin not found.' ), array( 'status' => 404 ) );
	}

	$data          = $plugins[ $plugin ];
	$data['_file'] = $plugin;

	return $data;
}
