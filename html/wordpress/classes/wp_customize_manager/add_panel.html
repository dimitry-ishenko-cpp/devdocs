public function add_panel( $id, $args = array() ) {
	if ( $id instanceof WP_Customize_Panel ) {
		$panel = $id;
	} else {
		$panel = new WP_Customize_Panel( $this, $id, $args );
	}

	$this->panels[ $panel->id ] = $panel;
	return $panel;
}
