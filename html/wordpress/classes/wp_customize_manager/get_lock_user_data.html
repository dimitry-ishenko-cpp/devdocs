protected function get_lock_user_data( $user_id ) {
	if ( ! $user_id ) {
		return null;
	}

	$lock_user = get_userdata( $user_id );

	if ( ! $lock_user ) {
		return null;
	}

	$user_details = array(
		'id'   => $lock_user->ID,
		'name' => $lock_user->display_name,
	);

	if ( get_option( 'show_avatars' ) ) {
		$user_details['avatar'] = get_avatar_url( $lock_user->ID, array( 'size' => 128 ) );
	}

	return $user_details;
}
