public function wp_die_handler() {
	_deprecated_function( __METHOD__, '4.7.0' );

	if ( $this->doing_ajax() || isset( $_POST['customized'] ) ) {
		return '_ajax_wp_die_handler';
	}

	return '_default_wp_die_handler';
}
