public function add_control( $id, $args = array() ) {
	if ( $id instanceof WP_Customize_Control ) {
		$control = $id;
	} else {
		$control = new WP_Customize_Control( $this, $id, $args );
	}

	$this->controls[ $control->id ] = $control;
	return $control;
}
