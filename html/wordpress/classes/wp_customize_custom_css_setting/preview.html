public function preview() {
	if ( $this->is_previewed ) {
		return false;
	}
	$this->is_previewed = true;
	add_filter( 'wp_get_custom_css', array( $this, 'filter_previewed_wp_get_custom_css' ), 9, 2 );
	return true;
}
