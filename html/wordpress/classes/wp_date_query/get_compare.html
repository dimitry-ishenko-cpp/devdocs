public function get_compare( $query ) {
	if ( ! empty( $query['compare'] )
		&& in_array( $query['compare'], array( '=', '!=', '>', '>=', '<', '<=', 'IN', 'NOT IN', 'BETWEEN', 'NOT BETWEEN' ), true )
	) {
		return strtoupper( $query['compare'] );
	}

	return $this->compare;
}
