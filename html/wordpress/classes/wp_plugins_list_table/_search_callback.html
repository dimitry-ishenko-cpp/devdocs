public function _search_callback( $plugin ) {
	global $s;

	foreach ( $plugin as $value ) {
		if ( is_string( $value ) && false !== stripos( strip_tags( $value ), urldecode( $s ) ) ) {
			return true;
		}
	}

	return false;
}
