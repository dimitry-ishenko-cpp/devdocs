function popstat () {
    //  Returns an array of 2 elements. The number of undeleted
    //  msgs in the mailbox, and the size of the mbox in octets.

    $PopArray = $this->last("array");

    if($PopArray == -1) { return false; }

    if( (!$PopArray) or (empty($PopArray)) )
    {
        return false;
    }
    return $PopArray;
}
