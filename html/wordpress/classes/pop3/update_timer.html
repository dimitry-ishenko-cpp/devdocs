POP3::update_timer()

Source

function update_timer () {
    if(function_exists("set_time_limit")){
        // Allows additional extension of POP3 request timeout to specified TIMEOUT property when update_timer is called.
        set_time_limit($this->TIMEOUT);
    }
    return true;
}
