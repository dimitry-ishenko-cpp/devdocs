public function __construct($message, $type, $data = null, $code = 0) {
	if ($type !== null) {
		$this->type = $type;
	}

	if ($code !== null) {
		$this->code = $code;
	}

	if ($message !== null) {
		$this->reason = $message;
	}

	$message = sprintf('%d %s', $this->code, $this->reason);
	parent::__construct($message, $this->type, $data, $this->code);
}
