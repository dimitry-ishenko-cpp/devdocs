public static function get_class($code) {
	if (!$code) {
		return 'Requests_Exception_HTTP_Unknown';
	}

	$class = sprintf('Requests_Exception_HTTP_%d', $code);
	if (class_exists($class)) {
		return $class;
	}

	return 'Requests_Exception_HTTP_Unknown';
}
