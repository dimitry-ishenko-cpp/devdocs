public function __construct($reason = null, $data = null) {
	if ($reason !== null) {
		$this->reason = $reason;
	}

	$message = sprintf('%d %s', $this->code, $this->reason);
	parent::__construct($message, 'httpresponse', $data, $this->code);
}
