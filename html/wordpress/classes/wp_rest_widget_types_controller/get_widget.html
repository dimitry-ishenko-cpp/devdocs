public function get_widget( $id ) {
	foreach ( $this->get_widgets() as $widget ) {
		if ( $id === $widget['id'] ) {
			return $widget;
		}
	}

	return new WP_Error( 'rest_widget_type_invalid', __( 'Invalid widget type.' ), array( 'status' => 404 ) );
}
