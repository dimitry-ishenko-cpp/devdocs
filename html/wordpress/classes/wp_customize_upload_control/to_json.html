public function to_json() {
	parent::to_json();

	$value = $this->value();
	if ( $value ) {
		// Get the attachment model for the existing file.
		$attachment_id = attachment_url_to_postid( $value );
		if ( $attachment_id ) {
			$this->json['attachment'] = wp_prepare_attachment_for_js( $attachment_id );
		}
	}
}
