protected function get_captured_option( $option_name, $default_value = false ) {
	if ( array_key_exists( $option_name, $this->_captured_options ) ) {
		$value = $this->_captured_options[ $option_name ];
	} else {
		$value = $default_value;
	}
	return $value;
}
