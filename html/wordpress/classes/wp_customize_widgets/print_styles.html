WP_Customize_Widgets::print_styles()

Calls admin_print_styles-widgets.php and admin_print_styles hooks to allow custom styles from plugins.

Source

public function print_styles() {
	/** This action is documented in wp-admin/admin-header.php */
	do_action( 'admin_print_styles-widgets.php' ); // phpcs:ignore WordPress.NamingConventions.ValidHookName.UseUnderscores

	/** This action is documented in wp-admin/admin-header.php */
	do_action( 'admin_print_styles' );
}
