public function __construct( $method = '', $route = '', $attributes = array() ) {
	$this->params = array(
		'URL'      => array(),
		'GET'      => array(),
		'POST'     => array(),
		'FILES'    => array(),

		// See parse_json_params.
		'JSON'     => null,

		'defaults' => array(),
	);

	$this->set_method( $method );
	$this->set_route( $route );
	$this->set_attributes( $attributes );
}
