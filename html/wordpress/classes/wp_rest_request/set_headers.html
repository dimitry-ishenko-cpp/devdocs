WP_REST_Request::set_headers( array $headers, bool $override = true )

Sets headers on the request.

Parameters

$headers array Required
Map of header name to value.
$override bool Optional
If true, replace the request's headers. Otherwise, merge with existing.

Default: true

Source

File: wp-includes/rest-api/class-wp-rest-request.php. View all references 

public function set_headers( $headers, $override = true ) {
	if ( true === $override ) {
		$this->headers = array();
	}

	foreach ( $headers as $key => $value ) {
		$this->set_header( $key, $value );
	}
}
