WP_REST_Request::set_body( string $data )

Sets body content.

Parameters

$data string Required
Binary data from the request body.

Source

public function set_body( $data ) {
	$this->body = $data;

	// Enable lazy parsing.
	$this->parsed_json    = false;
	$this->parsed_body    = false;
	$this->params['JSON'] = null;
}
