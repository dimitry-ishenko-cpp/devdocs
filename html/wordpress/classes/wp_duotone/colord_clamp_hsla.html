private static function colord_clamp_hsla( $hsla ) {
	$hsla['h'] = self::colord_clamp_hue( $hsla['h'] );
	$hsla['s'] = self::colord_clamp( $hsla['s'], 0, 100 );
	$hsla['l'] = self::colord_clamp( $hsla['l'], 0, 100 );
	$hsla['a'] = self::colord_clamp( $hsla['a'] );

	return $hsla;
}
