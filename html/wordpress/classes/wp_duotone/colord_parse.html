private static function colord_parse( $input ) {
	$result = self::colord_parse_hex( $input );

	if ( ! $result ) {
		$result = self::colord_parse_rgba_string( $input );
	}

	if ( ! $result ) {
		$result = self::colord_parse_hsla_string( $input );
	}

	return $result;
}
