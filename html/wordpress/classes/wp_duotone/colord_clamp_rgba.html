private static function colord_clamp_rgba( $rgba ) {
	$rgba['r'] = self::colord_clamp( $rgba['r'], 0, 255 );
	$rgba['g'] = self::colord_clamp( $rgba['g'], 0, 255 );
	$rgba['b'] = self::colord_clamp( $rgba['b'], 0, 255 );
	$rgba['a'] = self::colord_clamp( $rgba['a'] );

	return $rgba;
}
