private static function colord_hsla_to_hsva( $hsla ) {
	$h = $hsla['h'];
	$s = $hsla['s'];
	$l = $hsla['l'];
	$a = $hsla['a'];

	$s *= ( $l < 50 ? $l : 100 - $l ) / 100;

	return array(
		'h' => $h,
		's' => $s > 0 ? ( ( 2 * $s ) / ( $l + $s ) ) * 100 : 0,
		'v' => $l + $s,
		'a' => $a,
	);
}
