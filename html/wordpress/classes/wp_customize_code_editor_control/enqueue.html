public function enqueue() {
	$this->editor_settings = wp_enqueue_code_editor(
		array_merge(
			array(
				'type'       => $this->code_type,
				'codemirror' => array(
					'indentUnit' => 2,
					'tabSize'    => 2,
				),
			),
			$this->editor_settings
		)
	);
}
