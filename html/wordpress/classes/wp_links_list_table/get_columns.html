public function get_columns() {
	return array(
		'cb'         => '<input type="checkbox" />',
		'name'       => _x( 'Name', 'link name' ),
		'url'        => __( 'URL' ),
		'categories' => __( 'Categories' ),
		'rel'        => __( 'Relationship' ),
		'visible'    => __( 'Visible' ),
		'rating'     => __( 'Rating' ),
	);
}
