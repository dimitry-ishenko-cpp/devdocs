WP_Sitemaps_Registry::get_provider( string $name ): WP_Sitemaps_Provider|null

Returns a single registered sitemap provider.

Parameters

$name string Required
Sitemap provider name.

Return

WP_Sitemaps_Provider|null Sitemap provider if it exists, null otherwise.

public function get_provider( $name ) {
	if ( ! is_string( $name ) || ! isset( $this->providers[ $name ] ) ) {
		return null;
	}

	return $this->providers[ $name ];
}
