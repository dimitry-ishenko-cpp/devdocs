private static function to_kebab_case( array $data ) {
	foreach ( $data as $key => $value ) {
		$kebab_case          = _wp_to_kebab_case( $key );
		$data[ $kebab_case ] = $value;
		if ( $kebab_case !== $key ) {
			unset( $data[ $key ] );
		}
	}

	return $data;
}
