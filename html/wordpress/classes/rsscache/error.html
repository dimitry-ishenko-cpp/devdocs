function error ($errormsg, $lvl=E_USER_WARNING) {
	$this->ERROR = $errormsg;
	if ( MAGPIE_DEBUG ) {
		wp_trigger_error( '', $errormsg, $lvl);
	}
	else {
		error_log( $errormsg, 0);
	}
}
