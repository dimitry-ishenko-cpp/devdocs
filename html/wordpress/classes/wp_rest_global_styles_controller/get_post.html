protected function get_post( $id ) {
	$error = new WP_Error(
		'rest_global_styles_not_found',
		__( 'No global styles config exist with that id.' ),
		array( 'status' => 404 )
	);

	$id = (int) $id;
	if ( $id <= 0 ) {
		return $error;
	}

	$post = get_post( $id );
	if ( empty( $post ) || empty( $post->ID ) || $this->post_type !== $post->post_type ) {
		return $error;
	}

	return $post;
}
