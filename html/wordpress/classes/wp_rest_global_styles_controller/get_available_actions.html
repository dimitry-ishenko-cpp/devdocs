protected function get_available_actions() {
	$rels = array();

	$post_type = get_post_type_object( $this->post_type );
	if ( current_user_can( $post_type->cap->publish_posts ) ) {
		$rels[] = 'https://api.w.org/action-publish';
	}

	return $rels;
}
