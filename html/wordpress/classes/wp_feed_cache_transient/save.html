public function save( $data ) {
	if ( $data instanceof SimplePie\SimplePie ) {
		$data = $data->data;
	}

	set_transient( $this->name, $data, $this->lifetime );
	set_transient( $this->mod_name, time(), $this->lifetime );
	return true;
}
