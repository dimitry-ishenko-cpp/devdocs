public function no_items() {
	global $comment_status;

	if ( 'moderated' === $comment_status ) {
		_e( 'No comments awaiting moderation.' );
	} elseif ( 'trash' === $comment_status ) {
		_e( 'No comments found in Trash.' );
	} else {
		_e( 'No comments found.' );
	}
}
