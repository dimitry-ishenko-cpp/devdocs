public function offsetSet($offset, $value) {
	if ($offset === null) {
		throw new Exception('Object is a dictionary, not a list', 'invalidset');
	}

	if (is_string($offset)) {
		$offset = strtolower($offset);
	}

	if (!isset($this->data[$offset])) {
		$this->data[$offset] = [];
	}

	$this->data[$offset][] = $value;
}
