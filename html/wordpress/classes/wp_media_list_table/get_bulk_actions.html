protected function get_bulk_actions() {
	$actions = array();

	if ( MEDIA_TRASH ) {
		if ( $this->is_trash ) {
			$actions['untrash'] = __( 'Restore' );
			$actions['delete']  = __( 'Delete permanently' );
		} else {
			$actions['trash'] = __( 'Move to Trash' );
		}
	} else {
		$actions['delete'] = __( 'Delete permanently' );
	}

	if ( $this->detached ) {
		$actions['attach'] = __( 'Attach' );
	}

	return $actions;
}
