public function get_columns() {
	$columns = array(
		'cb'                => '<input type="checkbox" />',
		'email'             => __( 'Requester' ),
		'status'            => __( 'Status' ),
		'created_timestamp' => __( 'Requested' ),
		'next_steps'        => __( 'Next steps' ),
	);
	return $columns;
}
