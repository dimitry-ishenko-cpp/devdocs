protected static function get_file_path_from_theme( $file_name, $template = false ) {
	$path      = $template ? get_template_directory() : get_stylesheet_directory();
	$candidate = $path . '/' . $file_name;

	return is_readable( $candidate ) ? $candidate : '';
}
