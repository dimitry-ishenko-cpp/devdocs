private static function remove_json_comments( $array ) {
	unset( $array['//'] );
	foreach ( $array as $k => $v ) {
		if ( is_array( $v ) ) {
			$array[ $k ] = static::remove_json_comments( $v );
		}
	}

	return $array;
}
