WP_Upgrader_Skin::header()

Displays the header before the update process.

Source

public function header() {
	if ( $this->done_header ) {
		return;
	}
	$this->done_header = true;
	echo '<div class="wrap">';
	echo '<h1>' . $this->options['title'] . '</h1>';
}
