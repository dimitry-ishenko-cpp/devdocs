protected function _sort_menus_by_orderby( $menu1, $menu2 ) {
	_deprecated_function( __METHOD__, '4.7.0', 'wp_list_sort' );

	$key = $this->_current_menus_sort_orderby;
	return strcmp( $menu1->$key, $menu2->$key );
}
