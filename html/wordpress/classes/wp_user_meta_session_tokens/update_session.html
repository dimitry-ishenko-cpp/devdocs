WP_User_Meta_Session_Tokens::update_session( string $verifier, array $session = null )

Updates a session based on its verifier (token hash).

Parameters

$verifier string Required
Verifier for the session to update.
$session array Optional
Session. Omitting this argument destroys the session.

Default: null

protected function update_session( $verifier, $session = null ) {
	$sessions = $this->get_sessions();

	if ( $session ) {
		$sessions[ $verifier ] = $session;
	} else {
		unset( $sessions[ $verifier ] );
	}

	$this->update_sessions( $sessions );
}
