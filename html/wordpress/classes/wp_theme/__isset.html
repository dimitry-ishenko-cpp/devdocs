WP_Theme::__isset( string $offset ): bool

__isset() magic method for properties formerly returned by current_theme_info()

Parameters

$offset string Required
Property to check if set.

Return

bool Whether the given property is set.

File: wp-includes/class-wp-theme.php. View all references 

public function __isset( $offset ) {
	static $properties = array(
		'name',
		'title',
		'version',
		'parent_theme',
		'template_dir',
		'stylesheet_dir',
		'template',
		'stylesheet',
		'screenshot',
		'description',
		'author',
		'tags',
		'theme_root',
		'theme_root_uri',
	);

	return in_array( $offset, $properties, true );
}
