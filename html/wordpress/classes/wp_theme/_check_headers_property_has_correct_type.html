private static function _check_headers_property_has_correct_type( $headers ) {
	if ( ! is_array( $headers ) ) {
		return false;
	}
	foreach ( $headers as $key => $value ) {
		if ( ! is_string( $key ) || ! is_string( $value ) ) {
			return false;
		}
	}
	return true;
}
