protected function check_create_permission( $post ) {
	$post_type = get_post_type_object( $post->post_type );

	if ( ! $this->check_is_post_type_allowed( $post_type ) ) {
		return false;
	}

	return current_user_can( $post_type->cap->create_posts );
}
