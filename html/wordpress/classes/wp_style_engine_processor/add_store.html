public function add_store( $store ) {
	if ( ! $store instanceof WP_Style_Engine_CSS_Rules_Store ) {
		_doing_it_wrong(
			__METHOD__,
			__( '$store must be an instance of WP_Style_Engine_CSS_Rules_Store' ),
			'6.1.0'
		);
		return $this;
	}

	$this->stores[ $store->get_name() ] = $store;

	return $this;
}
