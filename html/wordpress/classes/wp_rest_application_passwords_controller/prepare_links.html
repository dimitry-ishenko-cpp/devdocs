protected function prepare_links( WP_User $user, $item ) {
	return array(
		'self' => array(
			'href' => rest_url(
				sprintf(
					'%s/users/%d/application-passwords/%s',
					$this->namespace,
					$user->ID,
					$item['uuid']
				)
			),
		),
	);
}
