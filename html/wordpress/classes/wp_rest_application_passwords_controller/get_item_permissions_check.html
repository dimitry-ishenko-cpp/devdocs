WP_REST_Application_Passwords_Controller::get_item_permissions_check( WP_REST_Request $request ): true|WP_Error

Checks if a given request has access to get a specific application password.

Parameters

$request WP_REST_Request Required

Return

true|WP_Error True if the request has read access for the item, WP_Error object otherwise.

public function get_item_permissions_check( $request ) {
	$user = $this->get_user( $request );

	if ( is_wp_error( $user ) ) {
		return $user;
	}

	if ( ! current_user_can( 'read_app_password', $user->ID, $request['uuid'] ) ) {
		return new WP_Error(
			'rest_cannot_read_application_password',
			__( 'Sorry, you are not allowed to read this application password.' ),
			array( 'status' => rest_authorization_required_code() )
		);
	}

	return true;
}
