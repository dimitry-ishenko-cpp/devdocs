public function delete_items( $request ) {
	$user = $this->get_user( $request );

	if ( is_wp_error( $user ) ) {
		return $user;
	}

	$deleted = WP_Application_Passwords::delete_all_application_passwords( $user->ID );

	if ( is_wp_error( $deleted ) ) {
		return $deleted;
	}

	return new WP_REST_Response(
		array(
			'deleted' => true,
			'count'   => $deleted,
		)
	);
}
