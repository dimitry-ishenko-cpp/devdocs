protected static function filter_declaration( $property, $value, $spacer = '' ) {
	$filtered_value = wp_strip_all_tags( $value, true );
	if ( '' !== $filtered_value ) {
		return safecss_filter_attr( "{$property}:{$spacer}{$filtered_value}" );
	}
	return '';
}
