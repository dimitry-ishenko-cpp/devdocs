WP_Widget_Links::__construct()

Sets up a new Links widget instance.

Source

public function __construct() {
	$widget_ops = array(
		'description'                 => __( 'Your blogroll' ),
		'customize_selective_refresh' => true,
	);
	parent::__construct( 'links', __( 'Links' ), $widget_ops );
}
