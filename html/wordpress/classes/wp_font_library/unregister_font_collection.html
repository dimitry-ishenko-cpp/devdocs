public function unregister_font_collection( string $slug ) {
	if ( ! $this->is_collection_registered( $slug ) ) {
		_doing_it_wrong(
			__METHOD__,
			/* translators: %s: Font collection slug. */
			sprintf( __( 'Font collection "%s" not found.' ), $slug ),
			'6.5.0'
		);
		return false;
	}
	unset( $this->collections[ $slug ] );
	return true;
}
