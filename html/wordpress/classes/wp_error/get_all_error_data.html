public function get_all_error_data( $code = '' ) {
	if ( empty( $code ) ) {
		$code = $this->get_error_code();
	}

	$data = array();

	if ( isset( $this->additional_data[ $code ] ) ) {
		$data = $this->additional_data[ $code ];
	}

	if ( isset( $this->error_data[ $code ] ) ) {
		$data[] = $this->error_data[ $code ];
	}

	return $data;
}
