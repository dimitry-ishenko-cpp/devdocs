public function add_data( $data, $code = '' ) {
	if ( empty( $code ) ) {
		$code = $this->get_error_code();
	}

	if ( isset( $this->error_data[ $code ] ) ) {
		$this->additional_data[ $code ][] = $this->error_data[ $code ];
	}

	$this->error_data[ $code ] = $data;
}
