public static function get_user_application_password( $user_id, $uuid ) {
	$passwords = static::get_user_application_passwords( $user_id );

	foreach ( $passwords as $password ) {
		if ( $password['uuid'] === $uuid ) {
			return $password;
		}
	}

	return null;
}
