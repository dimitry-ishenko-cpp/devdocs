protected function get_term( $id ) {
	$error = new WP_Error(
		'rest_term_invalid',
		__( 'Term does not exist.' ),
		array( 'status' => 404 )
	);

	if ( ! $this->check_is_taxonomy_allowed( $this->taxonomy ) ) {
		return $error;
	}

	if ( (int) $id <= 0 ) {
		return $error;
	}

	$term = get_term( (int) $id, $this->taxonomy );
	if ( empty( $term ) || $term->taxonomy !== $this->taxonomy ) {
		return $error;
	}

	return $term;
}
