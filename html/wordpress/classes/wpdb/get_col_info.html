wpdb::get_col_info( string $info_type = 'name', int $col_offset = -1 ): mixed

Retrieves column metadata from the last query.

Parameters

$info_type string Optional
Possible values include 'name', 'table', 'def', 'max_length', 'not_null', 'primary_key', 'multiple_key', 'unique_key', 'numeric', 'blob', 'type', 'unsigned', 'zerofill'. Default 'name'.

Default: 'name'

$col_offset int Optional
0: col name. 1: which table the col's in. 2: col's max length.
3: if the col is numeric. 4: col's type.

Default: -1

Return

mixed Column results.

Source

public function get_col_info( $info_type = 'name', $col_offset = -1 ) {
	$this->load_col_info();

	if ( $this->col_info ) {
		if ( -1 === $col_offset ) {
			$i         = 0;
			$new_array = array();
			foreach ( (array) $this->col_info as $col ) {
				$new_array[ $i ] = $col->{$info_type};
				$i++;
			}
			return $new_array;
		} else {
			return $this->col_info[ $col_offset ]->{$info_type};
		}
	}
}
