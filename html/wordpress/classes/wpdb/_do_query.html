private function _do_query( $query ) {
	if ( defined( 'SAVEQUERIES' ) && SAVEQUERIES ) {
		$this->timer_start();
	}

	if ( ! empty( $this->dbh ) ) {
		$this->result = mysqli_query( $this->dbh, $query );
	}

	++$this->num_queries;

	if ( defined( 'SAVEQUERIES' ) && SAVEQUERIES ) {
		$this->log_query(
			$query,
			$this->timer_stop(),
			$this->get_caller(),
			$this->time_start,
			array()
		);
	}
}
