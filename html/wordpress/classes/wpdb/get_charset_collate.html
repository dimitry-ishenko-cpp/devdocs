public function get_charset_collate() {
	$charset_collate = '';

	if ( ! empty( $this->charset ) ) {
		$charset_collate = "DEFAULT CHARACTER SET $this->charset";
	}
	if ( ! empty( $this->collate ) ) {
		$charset_collate .= " COLLATE $this->collate";
	}

	return $charset_collate;
}
