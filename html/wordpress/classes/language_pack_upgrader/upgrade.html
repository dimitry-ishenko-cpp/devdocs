public function upgrade( $update = false, $args = array() ) {
	if ( $update ) {
		$update = array( $update );
	}

	$results = $this->bulk_upgrade( $update, $args );

	if ( ! is_array( $results ) ) {
		return $results;
	}

	return $results[0];
}
