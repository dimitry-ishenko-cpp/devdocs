private function compile_src( array $value ) {
	$src = '';

	foreach ( $value as $item ) {
		$src .= ( 'data' === $item['format'] )
			? ", url({$item['url']})"
			: ", url('{$item['url']}') format('{$item['format']}')";
	}

	$src = ltrim( $src, ', ' );
	return $src;
}
