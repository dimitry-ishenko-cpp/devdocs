public function read( $bytes ) {
	$data        = $this->substr( $this->_str, $this->_pos, $bytes );
	$this->_pos += $bytes;
	if ( $this->strlen( $this->_str ) < $this->_pos ) {
		$this->_pos = $this->strlen( $this->_str );
	}
	return $data;
}
