protected function prepare_links( $post ) {
	// Entity meta.
	$links = parent::prepare_links( $post );

	return array(
		'self'       => $links['self'],
		'collection' => $links['collection'],
		'font_faces' => $this->prepare_font_face_links( $post->ID ),
	);
}
