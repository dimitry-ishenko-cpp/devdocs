public function render_stylesheet( $type ) {
	header( 'Content-Type: application/xml; charset=UTF-8' );

	if ( 'sitemap' === $type ) {
		// All content is escaped below.
		echo $this->get_sitemap_stylesheet();
	}

	if ( 'index' === $type ) {
		// All content is escaped below.
		echo $this->get_sitemap_index_stylesheet();
	}

	exit;
}
