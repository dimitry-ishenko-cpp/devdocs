public function get_base_dir( $base = '.', $verbose = false ) {
	_deprecated_function( __FUNCTION__, '2.7.0', 'WP_Filesystem_Base::abspath() or WP_Filesystem_Base::wp_*_dir()' );
	$this->verbose = $verbose;
	return $this->abspath();
}
