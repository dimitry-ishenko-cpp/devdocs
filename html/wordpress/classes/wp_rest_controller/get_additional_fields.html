WP_REST_Controller::get_additional_fields( string $object_type = null ): array

Retrieves all of the registered additional fields for a given object-type.

Parameters

$object_type string Optional
The object type.

Default: null

Return

array Registered additional fields (if any), empty array if none or if the object type could not be inferred.

Source

File: wp-includes/rest-api/endpoints/class-wp-rest-controller.php. View all references 

protected function get_additional_fields( $object_type = null ) {
	global $wp_rest_additional_fields;

	if ( ! $object_type ) {
		$object_type = $this->get_object_type();
	}

	if ( ! $object_type ) {
		return array();
	}

	if ( ! $wp_rest_additional_fields || ! isset( $wp_rest_additional_fields[ $object_type ] ) ) {
		return array();
	}

	return $wp_rest_additional_fields[ $object_type ];
}
