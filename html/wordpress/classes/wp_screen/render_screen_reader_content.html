public function render_screen_reader_content( $key = '', $tag = 'h2' ) {

	if ( ! isset( $this->_screen_reader_content[ $key ] ) ) {
		return;
	}
	echo "<$tag class='screen-reader-text'>" . $this->_screen_reader_content[ $key ] . "</$tag>";
}
