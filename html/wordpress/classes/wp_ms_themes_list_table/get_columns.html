public function get_columns() {
	$columns = array(
		'cb'          => '<input type="checkbox" />',
		'name'        => __( 'Theme' ),
		'description' => __( 'Description' ),
	);

	if ( $this->show_autoupdates ) {
		$columns['auto-updates'] = __( 'Automatic Updates' );
	}

	return $columns;
}
