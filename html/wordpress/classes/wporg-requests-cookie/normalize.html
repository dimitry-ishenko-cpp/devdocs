public function normalize() {
	foreach ($this->attributes as $key => $value) {
		$orig_value = $value;

		if (is_string($key)) {
			$value = $this->normalize_attribute($key, $value);
		}

		if ($value === null) {
			unset($this->attributes[$key]);
			continue;
		}

		if ($value !== $orig_value) {
			$this->attributes[$key] = $value;
		}
	}

	return true;
}
