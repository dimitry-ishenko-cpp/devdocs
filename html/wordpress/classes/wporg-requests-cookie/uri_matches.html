public function uri_matches(Iri $uri) {
	if (!$this->domain_matches($uri->host)) {
		return false;
	}

	if (!$this->path_matches($uri->path)) {
		return false;
	}

	return empty($this->attributes['secure']) || $uri->scheme === 'https';
}
