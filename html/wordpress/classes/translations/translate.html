public function translate( $singular, $context = null ) {
	$entry      = new Translation_Entry(
		array(
			'singular' => $singular,
			'context'  => $context,
		)
	);
	$translated = $this->translate_entry( $entry );
	return ( $translated && ! empty( $translated->translations ) ) ? $translated->translations[0] : $singular;
}
