public function add_entry( $entry ) {
	if ( is_array( $entry ) ) {
		$entry = new Translation_Entry( $entry );
	}
	$key = $entry->key();
	if ( false === $key ) {
		return false;
	}
	$this->entries[ $key ] = &$entry;
	return true;
}
