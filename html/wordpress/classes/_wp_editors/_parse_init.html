private static function _parse_init( $init ) {
	$options = '';

	foreach ( $init as $key => $value ) {
		if ( is_bool( $value ) ) {
			$val      = $value ? 'true' : 'false';
			$options .= $key . ':' . $val . ',';
			continue;
		} elseif ( ! empty( $value ) && is_string( $value ) && (
			( '{' === $value[0] && '}' === $value[ strlen( $value ) - 1 ] ) ||
			( '[' === $value[0] && ']' === $value[ strlen( $value ) - 1 ] ) ||
			preg_match( '/^\(?function ?\(/', $value ) ) ) {

			$options .= $key . ':' . $value . ',';
			continue;
		}
		$options .= $key . ':"' . $value . '",';
	}

	return '{' . trim( $options, ' ,' ) . '}';
}
