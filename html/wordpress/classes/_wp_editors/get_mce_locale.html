public static function get_mce_locale() {
	if ( empty( self::$mce_locale ) ) {
		$mce_locale       = get_user_locale();
		self::$mce_locale = empty( $mce_locale ) ? 'en' : strtolower( substr( $mce_locale, 0, 2 ) ); // ISO 639-1.
	}

	return self::$mce_locale;
}
