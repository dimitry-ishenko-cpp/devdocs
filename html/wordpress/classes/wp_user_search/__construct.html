function __construct( $search_term = '', $page = '', $role = '' ) {
	_deprecated_function( __FUNCTION__, '3.1.0', 'WP_User_Query' );

	$this->search_term = wp_unslash( $search_term );
	$this->raw_page = ( '' == $page ) ? false : (int) $page;
	$this->page = ( '' == $page ) ? 1 : (int) $page;
	$this->role = $role;

	$this->prepare_query();
	$this->query();
	$this->do_paging();
}
