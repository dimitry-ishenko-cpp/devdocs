public function nplurals_and_expression_from_header( $header ) {
	if ( preg_match( '/^\s*nplurals\s*=\s*(\d+)\s*;\s+plural\s*=\s*(.+)$/', $header, $matches ) ) {
		$nplurals   = (int) $matches[1];
		$expression = trim( $matches[2] );
		return array( $nplurals, $expression );
	} else {
		return array( 2, 'n != 1' );
	}
}
