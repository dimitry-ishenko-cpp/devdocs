Return

Source

public function get_data( $url, $args = '' ) {
	$args = wp_parse_args( $args );

	$provider = $this->get_provider( $url, $args );

	if ( ! $provider ) {
		return false;
	}

	$data = $this->fetch( $provider, $url, $args );

	if ( false === $data ) {
		return false;
	}

	return $data;
}
