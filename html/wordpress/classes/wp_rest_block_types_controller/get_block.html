protected function get_block( $name ) {
	$block_type = $this->block_registry->get_registered( $name );
	if ( empty( $block_type ) ) {
		return new WP_Error( 'rest_block_type_invalid', __( 'Invalid block type.' ), array( 'status' => 404 ) );
	}

	return $block_type;
}
