public function get_columns() {
	return array(
		'name'      => __( 'Name' ),
		'created'   => __( 'Created' ),
		'last_used' => __( 'Last Used' ),
		'last_ip'   => __( 'Last IP' ),
		'revoke'    => __( 'Revoke' ),
	);
}
