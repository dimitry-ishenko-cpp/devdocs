protected function get_search_sql( $search, $columns ) {
	global $wpdb;

	$like = '%' . $wpdb->esc_like( $search ) . '%';

	$searches = array();
	foreach ( $columns as $column ) {
		$searches[] = $wpdb->prepare( "$column LIKE %s", $like );
	}

	return ' AND (' . implode( ' OR ', $searches ) . ')';
}
