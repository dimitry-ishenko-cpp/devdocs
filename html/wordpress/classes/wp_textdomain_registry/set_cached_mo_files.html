private function set_cached_mo_files( $path ) {
	$this->cached_mo_files[ $path ] = array();

	$mo_files = glob( $path . '/*.mo' );

	if ( $mo_files ) {
		$this->cached_mo_files[ $path ] = $mo_files;
	}
}
