public function in_default_dir( $src ) {
	if ( ! $this->default_dirs ) {
		return true;
	}

	if ( str_starts_with( $src, '/' . WPINC . '/js/l10n' ) ) {
		return false;
	}

	foreach ( (array) $this->default_dirs as $test ) {
		if ( str_starts_with( $src, $test ) ) {
			return true;
		}
	}
	return false;
}
