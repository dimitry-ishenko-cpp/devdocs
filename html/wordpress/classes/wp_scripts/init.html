public function init() {
	/**
	 * Fires when the WP_Scripts instance is initialized.

	 * @since 2.6.0
	 *
	 * @param WP_Scripts $wp_scripts WP_Scripts instance (passed by reference).

	do_action_ref_array( 'wp_default_scripts', array( &$this ) );
