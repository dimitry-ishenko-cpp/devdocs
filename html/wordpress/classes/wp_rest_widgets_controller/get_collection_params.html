Source

public function get_collection_params() {
	return array(
		'context' => $this->get_context_param( array( 'default' => 'view' ) ),
		'sidebar' => array(
			'description' => __( 'The sidebar to return widgets for.' ),
			'type'        => 'string',
		),
	);
}
