public function column_registered( $user ) {
	global $mode;
	if ( 'list' === $mode ) {
		$date = __( 'Y/m/d' );
	} else {
		$date = __( 'Y/m/d g:i:s a' );
	}
	echo mysql2date( $date, $user->user_registered );
}
