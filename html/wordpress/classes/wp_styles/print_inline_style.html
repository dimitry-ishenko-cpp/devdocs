public function print_inline_style( $handle, $display = true ) {
	$output = $this->get_data( $handle, 'after' );

	if ( empty( $output ) ) {
		return false;
	}

	$output = implode( "\n", $output );

	if ( ! $display ) {
		return $output;
	}

	printf(
		"<style id='%s-inline-css'%s>\n%s\n</style>\n",
		esc_attr( $handle ),
		$this->type_attr,
		$output
	);

	return true;
}
