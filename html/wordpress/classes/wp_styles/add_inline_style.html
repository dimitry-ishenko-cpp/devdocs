public function add_inline_style( $handle, $code ) {
	if ( ! $code ) {
		return false;
	}

	$after = $this->get_data( $handle, 'after' );
	if ( ! $after ) {
		$after = array();
	}

	$after[] = $code;

	return $this->add_data( $handle, 'after', $after );
}
