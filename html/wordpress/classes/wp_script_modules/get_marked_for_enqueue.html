WP_Script_Modules::get_marked_for_enqueue(): array[]

Retrieves the list of script modules marked for enqueue.

Return

array[] Script modules marked for enqueue, keyed by script module identifier.

Source

private function get_marked_for_enqueue(): array {
	$enqueued = array();
	foreach ( $this->registered as $id => $script_module ) {
		if ( true === $script_module['enqueue'] ) {
			$enqueued[ $id ] = $script_module;
		}
	}
	return $enqueued;
}
