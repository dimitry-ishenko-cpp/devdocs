protected function get_max_batch_size() {
	/**
	 * Filters the maximum number of REST API requests that can be included in a batch.
	 *
	 * @since 5.6.0
	 *
	 * @param int $max_size The maximum size.
	 */
	return apply_filters( 'rest_get_max_batch_size', 25 );
}
