protected function add_image_to_index( WP_REST_Response $response, $image_id, $type ) {
	$response->data[ $type ] = (int) $image_id;
	if ( $image_id ) {
		$response->add_link(
			'https://api.w.org/featuredmedia',
			rest_url( rest_get_route_for_post( $image_id ) ),
			array(
				'embeddable' => true,
				'type'       => $type,
			)
		);
	}
}
