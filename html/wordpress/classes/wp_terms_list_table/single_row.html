public function single_row( $tag, $level = 0 ) {
	global $taxonomy;
	$tag = sanitize_term( $tag, $taxonomy );

	$this->level = $level;

	if ( $tag->parent ) {
		$count = count( get_ancestors( $tag->term_id, $taxonomy, 'taxonomy' ) );
		$level = 'level-' . $count;
	} else {
		$level = 'level-0';
	}

	echo '<tr id="tag-' . $tag->term_id . '" class="' . $level . '">';
	$this->single_row_columns( $tag );
	echo '</tr>';
}
