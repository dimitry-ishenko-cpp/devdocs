public static function should_decode( $headers ) {
	if ( is_array( $headers ) ) {
		if ( array_key_exists( 'content-encoding', $headers ) && ! empty( $headers['content-encoding'] ) ) {
			return true;
		}
	} elseif ( is_string( $headers ) ) {
		return ( stripos( $headers, 'content-encoding:' ) !== false );
	}

	return false;
}
