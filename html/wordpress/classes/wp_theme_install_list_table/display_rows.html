public function display_rows() {
	$themes = $this->items;
	foreach ( $themes as $theme ) {
		?>
			<div class="available-theme installable-theme">
			<?php
				$this->single_row( $theme );
			?>
			</div>
		<?php
	} // End foreach $theme_names.

	$this->theme_installer();
}
