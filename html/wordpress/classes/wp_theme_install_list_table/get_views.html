protected function get_views() {
	global $tabs, $tab;

	$display_tabs = array();
	foreach ( (array) $tabs as $action => $text ) {
		$display_tabs[ 'theme-install-' . $action ] = array(
			'url'     => self_admin_url( 'theme-install.php?tab=' . $action ),
			'label'   => $text,
			'current' => $action === $tab,
		);
	}

	return $this->get_views_links( $display_tabs );
}
