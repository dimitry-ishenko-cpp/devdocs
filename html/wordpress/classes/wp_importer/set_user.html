public function set_user( $user_id ) {
	if ( is_numeric( $user_id ) ) {
		$user_id = (int) $user_id;
	} else {
		$user_id = (int) username_exists( $user_id );
	}

	if ( ! $user_id || ! wp_set_current_user( $user_id ) ) {
		fwrite( STDERR, "Error: can not find user\n" );
		exit;
	}

	return $user_id;
}
