Checks if user has exceeded disk quota.

Return

bool

Source

public function is_user_over_quota() {
	if ( function_exists( 'upload_is_user_over_quota' ) ) {
		if ( upload_is_user_over_quota() ) {
			return true;
		}
	}

	return false;
}
