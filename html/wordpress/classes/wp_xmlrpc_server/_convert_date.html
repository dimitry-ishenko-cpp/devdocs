protected function _convert_date( $date ) {
	if ( '0000-00-00 00:00:00' === $date ) {
		return new IXR_Date( '00000000T00:00:00Z' );
	}
	return new IXR_Date( mysql2date( 'Ymd\TH:i:s', $date, false ) );
}
