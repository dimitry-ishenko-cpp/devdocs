wp_xmlrpc_server::get_term_custom_fields( int $term_id ): array

Parameters

Term ID.

Return

public function get_term_custom_fields( $term_id ) {
	$term_id = (int) $term_id;

	$custom_fields = array();

	foreach ( (array) has_term_meta( $term_id ) as $meta ) {

		if ( ! current_user_can( 'edit_term_meta', $term_id ) ) {
			continue;
		}

		$custom_fields[] = array(
			'id'    => $meta['meta_id'],
			'key'   => $meta['meta_key'],
			'value' => $meta['meta_value'],
		);
	}

	return $custom_fields;
}
