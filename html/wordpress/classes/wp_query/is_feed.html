public function is_feed( $feeds = '' ) {
	if ( empty( $feeds ) || ! $this->is_feed ) {
		return (bool) $this->is_feed;
	}

	$qv = $this->get( 'feed' );
	if ( 'feed' === $qv ) {
		$qv = get_default_feed();
	}

	return in_array( $qv, (array) $feeds, true );
}
