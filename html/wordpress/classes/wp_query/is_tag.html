WP_Query::is_tag( int|string|int[]|string[] $tag = '' ): bool

Is the query for an existing tag archive page?

Description

If the $tag parameter is specified, this function will additionally check if the query is for one of the tags specified.

Parameters

$tag int|string|int[]|string[] Optional
Tag ID, name, slug, or array of such to check against.

Default: ''

Return

bool Whether the query is for an existing tag archive page.

public function is_tag( $tag = '' ) {
	if ( ! $this->is_tag ) {
		return false;
	}

	if ( empty( $tag ) ) {
		return true;
	}

	$tag_obj = $this->get_queried_object();
	if ( ! $tag_obj ) {
		return false;
	}

	$tag = array_map( 'strval', (array) $tag );

	if ( in_array( (string) $tag_obj->term_id, $tag, true ) ) {
		return true;
	} elseif ( in_array( $tag_obj->name, $tag, true ) ) {
		return true;
	} elseif ( in_array( $tag_obj->slug, $tag, true ) ) {
		return true;
	}

	return false;
}
