Custom_Image_Header::js()

Execute JavaScript depending on step.

public function js() {
	$step = $this->step();

	if ( ( 1 === $step || 3 === $step ) && current_theme_supports( 'custom-header', 'header-text' ) ) {
		$this->js_1();
	} elseif ( 2 === $step ) {
		$this->js_2();
	}
}
