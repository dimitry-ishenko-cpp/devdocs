public function step() {
	if ( ! isset( $_GET['step'] ) ) {
		return 1;
	}

	$step = (int) $_GET['step'];
	if ( $step < 1 || 3 < $step ||
		( 2 === $step && ! wp_verify_nonce( $_REQUEST['_wpnonce-custom-header-upload'], 'custom-header-upload' ) ) ||
		( 3 === $step && ! wp_verify_nonce( $_REQUEST['_wpnonce'], 'custom-header-crop-image' ) )
	) {
		return 1;
	}

	return $step;
}
