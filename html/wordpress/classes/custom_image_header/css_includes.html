public function css_includes() {
	$step = $this->step();

	if ( ( 1 === $step || 3 === $step ) && current_theme_supports( 'custom-header', 'header-text' ) ) {
		wp_enqueue_style( 'wp-color-picker' );
	} elseif ( 2 === $step ) {
		wp_enqueue_style( 'imgareaselect' );
	}
}
