Custom_Image_Header::admin_page()

Displays the page based on the current step.

Source

public function admin_page() {
	if ( ! current_user_can( 'edit_theme_options' ) ) {
		wp_die( __( 'Sorry, you are not allowed to customize headers.' ) );
	}

	$step = $this->step();

	if ( 2 === $step ) {
		$this->step_2();
	} elseif ( 3 === $step ) {
		$this->step_3();
	} else {
		$this->step_1();
	}
}
