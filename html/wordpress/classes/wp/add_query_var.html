WP::add_query_var( string $qv )

Adds a query variable to the list of public query variables.

Parameters

$qvstringrequired
Query variable name.

Source

public function add_query_var( $qv ) {
	if ( ! in_array( $qv, $this->public_query_vars, true ) ) {
		$this->public_query_vars[] = $qv;
	}
}
