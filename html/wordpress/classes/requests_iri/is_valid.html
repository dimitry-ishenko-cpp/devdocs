public function is_valid() {
	$isauthority = $this->iuserinfo !== null || $this->ihost !== null || $this->port !== null;
	if ($this->ipath !== '' &&
		(
			$isauthority && $this->ipath[0] !== '/' ||
			(
				$this->scheme === null &&
				!$isauthority &&
				strpos($this->ipath, ':') !== false &&
				(strpos($this->ipath, '/') === false ? true : strpos($this->ipath, ':') < strpos($this->ipath, '/'))
			)
		)
	) {
		return false;
	}

	return true;
}
