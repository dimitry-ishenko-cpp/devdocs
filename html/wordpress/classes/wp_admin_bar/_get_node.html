final protected function _get_node( $id ) {
	if ( $this->bound ) {
		return;
	}

	if ( empty( $id ) ) {
		$id = 'root';
	}

	if ( isset( $this->nodes[ $id ] ) ) {
		return $this->nodes[ $id ];
	}
}
