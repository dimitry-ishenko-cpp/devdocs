Gets the error for an extension, if paused.

Parameters

Source

public function get( $extension ) {
	if ( ! $this->is_api_loaded() ) {
		return null;
	}

	$paused_extensions = $this->get_all();

	if ( ! isset( $paused_extensions[ $extension ] ) ) {
		return null;
	}

	return $paused_extensions[ $extension ];
}
