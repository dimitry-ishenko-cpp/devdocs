protected function get_option_name() {
	if ( ! wp_recovery_mode()->is_active() ) {
		return '';
	}

	$session_id = wp_recovery_mode()->get_session_id();
	if ( empty( $session_id ) ) {
		return '';
	}

	return "{$session_id}_paused_extensions";
}
