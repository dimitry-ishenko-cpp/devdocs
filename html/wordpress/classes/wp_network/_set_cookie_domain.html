private function _set_cookie_domain() {
	if ( ! empty( $this->cookie_domain ) ) {
		return;
	}

	$this->cookie_domain = $this->domain;
	if ( 'www.' === substr( $this->cookie_domain, 0, 4 ) ) {
		$this->cookie_domain = substr( $this->cookie_domain, 4 );
	}
}
