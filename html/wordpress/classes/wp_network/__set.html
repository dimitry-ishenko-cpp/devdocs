Description

public function __set( $key, $value ) {
	switch ( $key ) {
		case 'id':
			$this->id = (int) $value;
			break;
		case 'blog_id':
		case 'site_id':
			$this->blog_id = (string) $value;
			break;
		default:
			$this->$key = $value;
	}
}
