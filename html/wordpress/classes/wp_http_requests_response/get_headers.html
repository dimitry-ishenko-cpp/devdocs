public function get_headers() {
	// Ensure headers remain case-insensitive.
	$converted = new Requests_Utility_CaseInsensitiveDictionary();

	foreach ( $this->response->headers->getAll() as $key => $value ) {
		if ( count( $value ) === 1 ) {
			$converted[ $key ] = $value[0];
		} else {
			$converted[ $key ] = $value;
		}
	}

	return $converted;
}
