protected function is_network_plugin( $extension ) {
	if ( 'plugin' !== $extension['type'] ) {
		return false;
	}

	if ( ! is_multisite() ) {
		return false;
	}

	$network_plugins = wp_get_active_network_plugins();

	foreach ( $network_plugins as $plugin ) {
		if ( str_starts_with( $plugin, $extension['slug'] . '/' ) ) {
			return true;
		}
	}

	return false;
}
