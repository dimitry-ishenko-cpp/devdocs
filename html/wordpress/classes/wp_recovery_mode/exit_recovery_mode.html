public function exit_recovery_mode() {
	if ( ! $this->is_active() ) {
		return false;
	}

	$this->email_service->clear_rate_limit();
	$this->cookie_service->clear_cookie();

	wp_paused_plugins()->delete_all();
	wp_paused_themes()->delete_all();

	return true;
}
