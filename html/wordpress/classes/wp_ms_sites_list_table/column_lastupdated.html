public function column_lastupdated( $blog ) {
	global $mode;

	if ( 'list' === $mode ) {
		$date = __( 'Y/m/d' );
	} else {
		$date = __( 'Y/m/d g:i:s a' );
	}

	if ( '0000-00-00 00:00:00' === $blog['last_updated'] ) {
		_e( 'Never' );
	} else {
		echo mysql2date( $date, $blog['last_updated'] );
	}
}
