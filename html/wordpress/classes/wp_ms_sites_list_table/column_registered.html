public function column_registered( $blog ) {
	global $mode;

	if ( 'list' === $mode ) {
		$date = __( 'Y/m/d' );
	} else {
		$date = __( 'Y/m/d g:i:s a' );
	}

	if ( '0000-00-00 00:00:00' === $blog['registered'] ) {
		echo '&#x2014;';
	} else {
		echo mysql2date( $date, $blog['registered'] );
	}
}
