protected function prepare_links( $id ) {
	$links = array(
		'self'       => array(
			'href' => rest_url( rest_get_route_for_post( $id ) ),
		),
		'collection' => array(
			'href' => rest_url( rest_get_route_for_post_type_items( $this->post_type ) ),
		),
		'about'      => array(
			'href' => rest_url( 'wp/v2/types/' . $this->post_type ),
		),
	);

	return $links;
}
