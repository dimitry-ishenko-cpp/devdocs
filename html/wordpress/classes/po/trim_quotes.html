public static function trim_quotes( $s ) {
	if ( str_starts_with( $s, '"' ) ) {
		$s = substr( $s, 1 );
	}
	if ( str_ends_with( $s, '"' ) ) {
		$s = substr( $s, 0, -1 );
	}
	return $s;
}
