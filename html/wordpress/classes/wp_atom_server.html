class wp_atom_server {
	public function __call( $name, $arguments ) {
		_deprecated_function( __CLASS__ . '::' . $name, '3.5.0', 'the Atom Publishing Protocol plugin' );
	}

	public static function __callStatic( $name, $arguments ) {

	}
}
