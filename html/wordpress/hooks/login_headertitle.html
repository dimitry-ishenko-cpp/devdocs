apply_filters_deprecated( 'login_headertitle', string $login_header_title )

This hook has been deprecated. Use ‘login_headertext’ instead.

Filters the title attribute of the header logo above login form.

Parameters

$login_header_title string
Login header logo title attribute.

Source

File: wp-login.php. View all references 

$login_header_title = apply_filters_deprecated(
	'login_headertitle',
	array( $login_header_title ),
	'5.2.0',
	'login_headertext',
	__( 'Usage of the title attribute on the login logo is not recommended for accessibility reasons. Use the link text instead.' )
);
