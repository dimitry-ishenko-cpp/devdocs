apply_filters( 'network_edit_site_nav_links', array $links )

Filters the links that appear on site-editing network pages.

Description

Default links: ‘site-info’, ‘site-users’, ‘site-themes’, and ‘site-settings’.

Parameters

$links array
An array of link data representing individual network admin pages.
  • link_slugarray
    An array of information about the individual link to a page.
    $type string $label Label to use for the link.
    $type string $url URL, relative to network_admin_url() to use for the link.
    $type string $cap Capability required to see the link.
    }

    Source

    File: wp-admin/includes/ms.php. View all references 

    $links = apply_filters(
	'network_edit_site_nav_links',
	array(
		'site-info'     => array(
			'label' => __( 'Info' ),
			'url'   => 'site-info.php',
			'cap'   => 'manage_sites',
		),
		'site-users'    => array(
			'label' => __( 'Users' ),
			'url'   => 'site-users.php',
			'cap'   => 'manage_sites',
		),
		'site-themes'   => array(
			'label' => __( 'Themes' ),
			'url'   => 'site-themes.php',
			'cap'   => 'manage_sites',
		),
		'site-settings' => array(
			'label' => __( 'Settings' ),
			'url'   => 'site-settings.php',
			'cap'   => 'manage_sites',
		),
	)
);
