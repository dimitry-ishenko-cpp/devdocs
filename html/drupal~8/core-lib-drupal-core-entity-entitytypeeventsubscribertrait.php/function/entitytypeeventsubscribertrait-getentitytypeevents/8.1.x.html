public static function getEntityTypeEvents() {
  $event = array('onEntityTypeEvent', 100);
  $events[EntityTypeEvents::CREATE][] = $event;
  $events[EntityTypeEvents::UPDATE][] = $event;
  $events[EntityTypeEvents::DELETE][] = $event;
  return $events;
}
