public static function preRenderTel($element) {
  $element['#attributes']['type'] = 'tel';
  Element::setAttributes($element, array('id', 'name', 'value', 'size', 'maxlength', 'placeholder'));
  static::setAttributes($element, array('form-tel'));

  return $element;
}
