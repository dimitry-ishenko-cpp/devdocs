public function setDefault($name) {
  $list = $this->listInfo();
  if (!isset($list[$name])) {
    throw new \InvalidArgumentException("$name theme is not installed.");
  }
  $this->configFactory->getEditable('system.theme')
    ->set('default', $name)
    ->save();
  return $this;
}
