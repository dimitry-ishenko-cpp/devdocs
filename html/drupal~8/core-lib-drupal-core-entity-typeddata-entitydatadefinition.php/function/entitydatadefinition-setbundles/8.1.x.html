public function setBundles(array $bundles = NULL) {
  if (isset($bundles)) {
    $this->addConstraint('Bundle', $bundles);
  }
  else {
    // Remove the constraint.
    unset($this->definition['constraints']['Bundle']);
  }
  return $this;
}
