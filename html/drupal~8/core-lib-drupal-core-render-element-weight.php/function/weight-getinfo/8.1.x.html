public function getInfo() {
  $class = get_class($this);
  return array(
    '#input' => TRUE,
    '#delta' => 10,
    '#default_value' => 0,
    '#process' => array(
      array($class, 'processWeight'),
      array($class, 'processAjaxForm'),
    ),
  );
}
