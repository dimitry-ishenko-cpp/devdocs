public function aggregate($field, $function, $langcode = NULL, &$alias = NULL) {
  if (!isset($alias)) {
    $alias = $this->getAggregationAlias($field, $function);
  }

  $this->aggregate[$alias] = array(
    'field' => $field,
    'function' => $function,
    'alias' => $alias,
    'langcode' => $langcode,
  );

  return $this;
}
