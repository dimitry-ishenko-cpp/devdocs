public function listAll() {
  $names = array();
  if (file_exists($this->directory)) {
    foreach (new \DirectoryIterator($this->directory) as $fileinfo) {
      if (!$fileinfo->isDot()) {
        $name = $fileinfo->getFilename();
        if ($name != '.htaccess') {
          $names[] = $name;
        }
      }
    }
  }
  return $names;
}
