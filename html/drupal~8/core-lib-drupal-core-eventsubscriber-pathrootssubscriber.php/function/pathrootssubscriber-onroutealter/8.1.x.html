public function onRouteAlter(RouteBuildEvent $event) {
  $collection = $event->getRouteCollection();
  foreach ($collection->all() as $route) {
    $bits = explode('/', ltrim($route->getPath(), '/'));
    $this->pathRoots[$bits[0]] = $bits[0];
  }
}
