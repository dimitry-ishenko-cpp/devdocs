public function import() {
  if ($this->hasUnprocessedConfigurationChanges()) {
    $sync_steps = $this->initialize();

    foreach ($sync_steps as $step) {
      $context = array();
      do {
        $this->doSyncStep($step, $context);
      } while ($context['finished'] < 1);
    }
  }
  return $this;
}
