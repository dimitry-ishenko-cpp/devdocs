public function getInfo() {
  $class = get_class($this);
  return array(
    '#input' => TRUE,
    '#size' => 60,
    '#maxlength' => 128,
    '#autocomplete_route_name' => FALSE,
    '#process' => array(
      array($class, 'processAutocomplete'),
      array($class, 'processAjaxForm'),
    ),
    '#pre_render' => array(
      array($class, 'preRenderSearch'),
    ),
    '#theme' => 'input__search',
    '#theme_wrappers' => array('form_element'),
  );
}
