public function importDelete($name, Config $new_config, Config $old_config) {
  $id = static::getIDFromConfigName($name, $this->entityType->getConfigPrefix());
  $entity = $this->load($id);
  $entity->setSyncing(TRUE);
  $entity->delete();
  return TRUE;
}
