public function toLink($text = NULL, $rel = 'canonical', array $options = []) {
  if (!isset($text)) {
    $text = $this->label();
  }
  $url = $this->toUrl($rel);
  $options += $url->getOptions();
  $url->setOptions($options);
  return new Link($text, $url);
}
