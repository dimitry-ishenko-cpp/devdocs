public function getDependencies($type) {
  $dependencies = array();
  if (isset($this->dependencies[$type])) {
    $dependencies = $this->dependencies[$type];
  }
  if ($type == 'module') {
    $dependencies[] = substr($this->name, 0, strpos($this->name, '.'));
  }
  return $dependencies;
}
