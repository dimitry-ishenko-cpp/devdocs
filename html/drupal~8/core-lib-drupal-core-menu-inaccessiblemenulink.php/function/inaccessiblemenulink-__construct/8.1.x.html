public function __construct(MenuLinkInterface $wrapped_link) {
  $this->wrappedLink = $wrapped_link;
  $plugin_definition = [
    'route_name' => '<front>',
    'route_parameters' => [],
    'url' => NULL,
  ] + $this->wrappedLink->getPluginDefinition();
  parent::__construct([], $this->wrappedLink->getPluginId(), $plugin_definition);
}
