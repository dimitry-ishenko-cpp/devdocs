function tracker_node_predelete(EntityInterface $node, $arg = 0) {
  db_delete('tracker_node')
    ->condition('nid', $node->id())
    ->execute();
  db_delete('tracker_user')
    ->condition('nid', $node->id())
    ->execute();
}
