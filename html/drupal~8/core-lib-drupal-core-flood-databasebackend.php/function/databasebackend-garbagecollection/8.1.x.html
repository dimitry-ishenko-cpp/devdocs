public function garbageCollection() {
  try {
    $return = $this->connection->delete(static::TABLE_NAME)
      ->condition('expiration', REQUEST_TIME, '<')
      ->execute();
  }
  catch (\Exception $e) {
    $this->catchException($e);
  }
}
