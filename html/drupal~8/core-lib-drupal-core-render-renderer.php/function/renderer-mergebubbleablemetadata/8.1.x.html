public function mergeBubbleableMetadata(array $a, array $b) {
  $meta_a = BubbleableMetadata::createFromRenderArray($a);
  $meta_b = BubbleableMetadata::createFromRenderArray($b);
  $meta_a->merge($meta_b)->applyTo($a);
  return $a;
}
