protected function getByHash($cidhash, $allow_invalid = FALSE) {
  if ($file = $this->storage()->getFullPath($cidhash)) {
    $cache = @include $file;
  }
  if (isset($cache)) {
    return $this->prepareItem($cache, $allow_invalid);
  }
  return FALSE;
}
