public function deleteMultiple(array $keys) {
  // Delete in chunks when a large array is passed.
  while ($keys) {
    $this->connection->delete($this->table)
      ->condition('name', array_splice($keys, 0, 1000), 'IN')
      ->condition('collection', $this->collection)
      ->execute();
  }
}
