public function set($filepath, $data) {
  $realpath = realpath($filepath);
  $cached = [
    'mtime' => filemtime($filepath),
    'filepath' => $filepath,
    'data' => $data,
  ];

  $cid = $this->prefix . ':' . $this->collection . ':' . $realpath;
  static::$cached[$cid] = $cached;
  if ($this->cache) {
    $this->cache->store($cid, $cached);
  }
}
