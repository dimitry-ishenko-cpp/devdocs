public function delete($filepath) {
  $realpath = realpath($filepath);
  $cid = $this->prefix . ':' . $this->collection . ':' . $realpath;

  unset(static::$cached[$cid]);
  if ($this->cache) {
    $this->cache->delete($cid);
  }
}
