public function onRequest(GetResponseEvent $event) {
  // Pass to the Html class that the current request is an Ajax request.
  if ($event->getRequest()->request->get(static::AJAX_REQUEST_PARAMETER)) {
    Html::setIsAjax(TRUE);
  }
}
