public function __toString() {
  $return = '';
  /** @var \Drupal\Core\Template\AttributeValueBase $value */
  foreach ($this->storage as $name => $value) {
    $rendered = $value->render();
    if ($rendered) {
      $return .= ' ' . $rendered;
    }
  }
  return $return;
}
