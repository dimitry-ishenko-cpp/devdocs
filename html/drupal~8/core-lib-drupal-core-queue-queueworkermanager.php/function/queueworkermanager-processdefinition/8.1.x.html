public function processDefinition(&$definition, $plugin_id) {
  parent::processDefinition($definition, $plugin_id);

  // Assign a default time if a cron is specified.
  if (isset($definition['cron'])) {
    $definition['cron'] += [
      'time' => 15,
    ];
  }
}
