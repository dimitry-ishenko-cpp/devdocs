protected function getWrapper($scheme, $uri) {
  if (isset($this->info[$scheme]['service_id'])) {
    $instance = $this->container->get($this->info[$scheme]['service_id']);
    $instance->setUri($uri);
    return $instance;
  }

  return FALSE;
}
