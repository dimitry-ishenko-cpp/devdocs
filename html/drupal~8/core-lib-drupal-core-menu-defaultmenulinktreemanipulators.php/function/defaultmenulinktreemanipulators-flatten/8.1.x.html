public function flatten(array $tree) {
  foreach ($tree as $key => $element) {
    if ($tree[$key]->subtree) {
      $tree += $this->flatten($tree[$key]->subtree);
    }
    $tree[$key]->subtree = array();
  }
  return $tree;
}
