protected function getInstallArgs($overrides = array()) {
  $args = array(
    'make_backup' => FALSE,
    'install_dir' => $this->getInstallDirectory(),
    'backup_dir' => $this->getBackupDir(),
  );
  return array_merge($args, $overrides);
}
