protected function actions(array $form, FormStateInterface $form_state) {
  return array(
    'submit' => array(
      '#type' => 'submit',
      '#value' => $this->getConfirmText(),
      '#submit' => array(
        array($this, 'submitForm'),
      ),
    ),
    'cancel' => ConfirmFormHelper::buildCancelLink($this, $this->getRequest()),
  );
}
