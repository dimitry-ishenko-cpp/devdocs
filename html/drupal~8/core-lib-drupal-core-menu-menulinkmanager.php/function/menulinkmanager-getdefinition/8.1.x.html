public function getDefinition($plugin_id, $exception_on_invalid = TRUE) {
  $definition = $this->treeStorage->load($plugin_id);
  if (empty($definition) && $exception_on_invalid) {
    throw new PluginNotFoundException($plugin_id);
  }
  return $definition;
}
