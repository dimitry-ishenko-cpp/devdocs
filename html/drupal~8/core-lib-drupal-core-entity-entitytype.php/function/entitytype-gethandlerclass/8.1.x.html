public function getHandlerClass($handler_type, $nested = FALSE) {
  if ($this->hasHandlerClass($handler_type, $nested)) {
    $handlers = $this->getHandlerClasses();
    return $nested ? $handlers[$handler_type][$nested] : $handlers[$handler_type];
  }
}
