public function get($property) {
  if (property_exists($this, $property)) {
    $value = isset($this->{$property}) ? $this->{$property} : NULL;
  }
  else {
    $value = isset($this->additional[$property]) ? $this->additional[$property] : NULL;
  }
  return $value;
}
