protected function getCid() {
  if (!isset($this->cid)) {
    $route_parameters = $this->routeMatch->getRawParameters()->all();
    ksort($route_parameters);
    return 'active-trail:route:' . $this->routeMatch->getRouteName() . ':route_parameters:' . serialize($route_parameters);
  }

  return $this->cid;
}
