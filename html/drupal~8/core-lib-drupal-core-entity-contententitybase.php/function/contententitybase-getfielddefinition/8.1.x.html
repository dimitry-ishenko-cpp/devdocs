public function getFieldDefinition($name) {
  if (!isset($this->fieldDefinitions)) {
    $this->getFieldDefinitions();
  }
  if (isset($this->fieldDefinitions[$name])) {
    return $this->fieldDefinitions[$name];
  }
}
