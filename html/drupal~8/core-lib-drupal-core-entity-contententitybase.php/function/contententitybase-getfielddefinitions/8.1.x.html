public function getFieldDefinitions() {
  if (!isset($this->fieldDefinitions)) {
    $this->fieldDefinitions = $this->entityManager()->getFieldDefinitions($this->entityTypeId, $this->bundle());
  }
  return $this->fieldDefinitions;
}
