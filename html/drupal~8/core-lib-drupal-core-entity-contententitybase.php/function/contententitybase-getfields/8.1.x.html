public function getFields($include_computed = TRUE) {
  $fields = array();
  foreach ($this->getFieldDefinitions() as $name => $definition) {
    if ($include_computed || !$definition->isComputed()) {
      $fields[$name] = $this->get($name);
    }
  }
  return $fields;
}
