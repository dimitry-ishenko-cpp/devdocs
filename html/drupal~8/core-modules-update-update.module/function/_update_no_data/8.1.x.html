function _update_no_data() {
  $destination = \Drupal::destination()->getAsArray();
  return t('No update information available. <a href=":run_cron.html">Run cron</a> or <a href=":check_manually.html">check manually</a>.', array(
    ':run_cron' => \Drupal::url('system.run_cron', [], ['query' => $destination]),
    ':check_manually' => \Drupal::url('update.manual_status', [], ['query' => $destination]),
  ));
}
