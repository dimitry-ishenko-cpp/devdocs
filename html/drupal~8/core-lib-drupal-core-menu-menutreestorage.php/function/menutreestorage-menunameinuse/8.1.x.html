public function menuNameInUse($menu_name) {
  $query = $this->connection->select($this->table, $this->options);
  $query->addField($this->table, 'mlid');
  $query->condition('menu_name', $menu_name);
  $query->range(0, 1);
  return (bool) $this->safeExecuteSelect($query);
}
