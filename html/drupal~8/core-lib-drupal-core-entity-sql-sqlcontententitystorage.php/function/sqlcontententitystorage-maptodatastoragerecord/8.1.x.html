protected function mapToDataStorageRecord(EntityInterface $entity, $table_name = NULL) {
  if (!isset($table_name)) {
    $table_name = $this->dataTable;
  }
  $record = $this->mapToStorageRecord($entity, $table_name);
  return $record;
}
