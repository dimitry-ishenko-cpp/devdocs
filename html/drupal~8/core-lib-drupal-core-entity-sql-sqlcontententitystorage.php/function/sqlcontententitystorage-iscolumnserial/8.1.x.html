protected function isColumnSerial($table_name, $schema_name) {
  $result = FALSE;

  switch ($table_name) {
    case $this->baseTable:
      $result = $schema_name == $this->idKey;
      break;

    case $this->revisionTable:
      $result = $schema_name == $this->revisionKey;
      break;
  }

  return $result;
}
