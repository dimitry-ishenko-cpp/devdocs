protected function getEntityIds() {
  $query = $this->getStorage()->getQuery()
    ->sort($this->entityType->getKey('id'));

  // Only add the pager if a limit is specified.
  if ($this->limit) {
    $query->pager($this->limit);
  }
  return $query->execute();
}
