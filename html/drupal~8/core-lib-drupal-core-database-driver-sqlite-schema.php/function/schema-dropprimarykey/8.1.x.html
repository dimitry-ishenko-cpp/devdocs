public function dropPrimaryKey($table) {
  $old_schema = $this->introspectSchema($table);
  $new_schema = $old_schema;

  if (empty($new_schema['primary key'])) {
    return FALSE;
  }

  unset($new_schema['primary key']);
  $this->alterTable($table, $old_schema, $new_schema);
  return TRUE;
}
