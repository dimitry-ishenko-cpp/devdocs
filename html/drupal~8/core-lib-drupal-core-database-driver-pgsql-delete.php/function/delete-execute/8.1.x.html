public function execute() {
  $this->connection->addSavepoint();
  try {
    $result = parent::execute();
  }
  catch (\Exception $e) {
    $this->connection->rollbackSavepoint();
    throw $e;
  }
  $this->connection->releaseSavepoint();

  return $result;
}
