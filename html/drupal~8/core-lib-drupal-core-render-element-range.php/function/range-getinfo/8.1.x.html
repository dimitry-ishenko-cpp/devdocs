public function getInfo() {
  $info = parent::getInfo();
  $class = get_class($this);
  return array(
    '#min' => 0,
    '#max' => 100,
    '#pre_render' => array(
      array($class, 'preRenderRange'),
    ),
    '#theme' => 'input__range',
  ) + $info;
}
