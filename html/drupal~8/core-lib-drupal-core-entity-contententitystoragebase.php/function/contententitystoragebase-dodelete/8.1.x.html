protected function doDelete($entities) {
  /** @var \Drupal\Core\Entity\ContentEntityInterface[] $entities */
  foreach ($entities as $entity) {
    $this->invokeFieldMethod('delete', $entity);
  }
  $this->doDeleteFieldItems($entities);
}
