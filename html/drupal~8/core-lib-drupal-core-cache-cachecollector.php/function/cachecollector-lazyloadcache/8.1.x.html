protected function lazyLoadCache() {
  if ($this->cacheLoaded) {
    return;
  }
  // The cache was not yet loaded, set flag to TRUE.
  $this->cacheLoaded = TRUE;

  if ($cache = $this->cache->get($this->getCid())) {
    $this->cacheCreated = $cache->created;
    $this->storage = $cache->data;
  }
}
