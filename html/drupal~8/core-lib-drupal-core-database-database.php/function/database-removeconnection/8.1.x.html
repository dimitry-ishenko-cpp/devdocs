final public static function removeConnection($key) {
  if (isset(self::$databaseInfo[$key])) {
    self::closeConnection(NULL, $key);
    unset(self::$databaseInfo[$key]);
    return TRUE;
  }
  else {
    return FALSE;
  }
}
