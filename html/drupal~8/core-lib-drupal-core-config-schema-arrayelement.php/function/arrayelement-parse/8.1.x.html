protected function parse() {
  $elements = array();
  foreach ($this->getAllKeys() as $key) {
    $value = isset($this->value[$key]) ? $this->value[$key] : NULL;
    $definition = $this->getElementDefinition($key);
    $elements[$key] = $this->createElement($definition, $value, $key);
  }
  return $elements;
}
