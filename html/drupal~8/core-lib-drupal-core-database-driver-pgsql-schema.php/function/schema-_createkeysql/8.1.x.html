protected function _createKeySql($fields) {
  $return = array();
  foreach ($fields as $field) {
    if (is_array($field)) {
      $return[] = 'substr(' . $field[0] . ', 1, ' . $field[1] . ')';
    }
    else {
      $return[] = '"' . $field . '"';
    }
  }
  return implode(', ', $return);
}
