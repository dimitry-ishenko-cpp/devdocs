protected function _createIndexSql($table, $name, $fields) {
  $query = 'CREATE INDEX ' . $this->ensureIdentifiersLength($table, $name, 'idx') . ' ON {' . $table . '} (';
  $query .= $this->_createKeySql($fields) . ')';
  return $query;
}
