function _color_hue2rgb($m1, $m2, $h) {
  $h = ($h < 0) ? $h + 1 : (($h > 1) ? $h - 1 : $h);
  if ($h * 6 < 1) {
    return $m1 + ($m2 - $m1) * $h * 6;
  }
  if ($h * 2 < 1) {
    return $m2;
  }
  if ($h * 3 < 2) {
    return $m1 + ($m2 - $m1) * (0.66666 - $h) * 6;
  }

  return $m1;
}
