protected function getModuleNamespacesPsr4($module_file_names) {
  $namespaces = array();
  foreach ($module_file_names as $module => $filename) {
    $namespaces["Drupal\\$module"] = dirname($filename) . '/src';
  }
  return $namespaces;
}
