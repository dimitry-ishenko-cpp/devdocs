protected function castSafeStrings($data) {
  if ($data instanceof MarkupInterface) {
    $data = (string) $data;
  }
  elseif (is_array($data)) {
    array_walk_recursive($data, function(&$value) {
      if ($value instanceof MarkupInterface) {
        $value = (string) $value;
      }
    });
  }
  return $data;
}
