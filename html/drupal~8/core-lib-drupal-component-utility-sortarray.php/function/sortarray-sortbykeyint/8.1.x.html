public static function sortByKeyInt($a, $b, $key) {
  $a_weight = (is_array($a) && isset($a[$key])) ? $a[$key] : 0;
  $b_weight = (is_array($b) && isset($b[$key])) ? $b[$key] : 0;

  if ($a_weight == $b_weight) {
    return 0;
  }

  return ($a_weight < $b_weight) ? -1 : 1;
}
