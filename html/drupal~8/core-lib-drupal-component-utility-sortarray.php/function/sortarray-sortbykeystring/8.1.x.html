public static function sortByKeyString($a, $b, $key) {
  $a_title = (is_array($a) && isset($a[$key])) ? $a[$key] : '';
  $b_title = (is_array($b) && isset($b[$key])) ? $b[$key] : '';

  return strnatcasecmp($a_title, $b_title);
}
