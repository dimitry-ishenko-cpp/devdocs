public function getRouteDebugMessage($name, array $parameters = array()) {
  if (is_scalar($name)) {
    return $name;
  }

  if ($name instanceof SymfonyRoute) {
    return 'Route with pattern ' . $name->getPath();
  }

  return serialize($name);
}
