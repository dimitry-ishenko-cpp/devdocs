protected function baseConfigurationDefaults() {
  return array(
    'id' => $this->getPluginId(),
    'label' => '',
    'provider' => $this->pluginDefinition['provider'],
    'label_display' => BlockInterface::BLOCK_LABEL_VISIBLE,
  );
}
