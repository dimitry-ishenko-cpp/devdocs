public function getInfo() {
  $class = get_class($this);
  return array(
    '#input' => TRUE,
    '#multiple' => FALSE,
    '#process' => array(
      array($class, 'processSelect'),
      array($class, 'processAjaxForm'),
    ),
    '#pre_render' => array(
      array($class, 'preRenderSelect'),
    ),
    '#theme' => 'select',
    '#theme_wrappers' => array('form_element'),
    '#options' => array(),
  );
}
