public function condition($field, $function = NULL, $value = NULL, $operator = NULL, $langcode = NULL) {
  $this->conditions[] = array(
    'field' => $field,
    'function' => $function,
    'value' => $value,
    'operator' => $operator,
    'langcode' => $langcode,
  );

  return $this;
