public function getInfo() {
  $class = get_class($this);
  return [
    '#input' => TRUE,
    '#theme' => 'input__date',
    '#process' => [[$class, 'processDate']],
    '#pre_render' => [[$class, 'preRenderDate']],
    '#theme_wrappers' => ['form_element'],
    '#attributes' => ['type' => 'date'],
    '#date_date_format' => 'Y-m-d',
  ];
}
