protected function processIdentifierSchema(&$schema, $key) {
  if ($schema['fields'][$key]['type'] == 'int') {
    $schema['fields'][$key]['type'] = 'serial';
  }
  $schema['fields'][$key]['not null'] = TRUE;
  unset($schema['fields'][$key]['default']);
}
