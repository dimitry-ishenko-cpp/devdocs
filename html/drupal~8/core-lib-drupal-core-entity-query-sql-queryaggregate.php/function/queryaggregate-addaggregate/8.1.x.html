protected function addAggregate() {
  if ($this->aggregate) {
    foreach ($this->aggregate as $aggregate) {
      $sql_field = $this->getSqlField($aggregate['field'], $aggregate['langcode']);
      $this->sqlExpressions[$aggregate['alias']] = $aggregate['function'] . "($sql_field)";
    }
  }
  return $this;
}
