protected function processHtmlHead(array $html_head) {
  $head = [];
  foreach ($html_head as $item) {
    list($data, $key) = $item;
    if (!isset($data['#type'])) {
      $data['#type'] = 'html_tag';
    }
    $head[$key] = $data;
  }
  return $head;
}
