public function resetCache(array $ids = NULL) {
  if ($this->entityType->isStaticallyCacheable() && isset($ids)) {
    foreach ($ids as $id) {
      unset($this->entities[$id]);
    }
  }
  else {
    $this->entities = array();
  }
}
