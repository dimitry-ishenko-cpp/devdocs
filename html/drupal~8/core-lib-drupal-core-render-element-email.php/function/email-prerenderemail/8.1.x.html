public static function preRenderEmail($element) {
  $element['#attributes']['type'] = 'email';
  Element::setAttributes($element, array('id', 'name', 'value', 'size', 'maxlength', 'placeholder'));
  static::setAttributes($element, array('form-email'));
  return $element;
}
