public function getRoles($exclude_locked_roles = FALSE) {
  $roles = $this->roles;

  if ($exclude_locked_roles) {
    $roles = array_values(array_diff($roles, array(AccountInterface::ANONYMOUS_ROLE, AccountInterface::AUTHENTICATED_ROLE)));
  }

  return $roles;
}
