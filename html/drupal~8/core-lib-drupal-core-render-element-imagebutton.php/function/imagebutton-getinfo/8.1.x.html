public function getInfo() {
  $info = parent::getInfo();
  unset($info['name']);

  return array(
    '#return_value' => TRUE,
    '#has_garbage_value' => TRUE,
    '#src' => NULL,
    '#theme_wrappers' => array('input__image_button'),
  ) + $info;
}
