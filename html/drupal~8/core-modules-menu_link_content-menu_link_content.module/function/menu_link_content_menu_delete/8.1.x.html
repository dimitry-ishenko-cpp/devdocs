function menu_link_content_menu_delete(MenuInterface $menu) {
  $storage = \Drupal::entityManager()->getStorage('menu_link_content');
  $menu_links = $storage->loadByProperties(array('menu_name' => $menu->id()));
  $storage->delete($menu_links);
}
