def self.create(arg)
  case arg
  when Alt
    return arg
  when String
    altitude = ''
    if Regex =~ arg
      altitude = [($1.to_f*(1e2))+(1e7)].pack("N")
    else
      raise ArgumentError.new("not a properly formed Alt string: " + arg)
    end
    return Alt.new(altitude)
  else
    raise ArgumentError.new("cannot interpret as Alt: #{arg.inspect}")
  end
end
