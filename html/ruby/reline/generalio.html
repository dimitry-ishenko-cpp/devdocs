getc(_timeout_second) Show source 
# File lib/reline/general_io.rb, line 41
def self.getc(_timeout_second)
  unless @@buf.empty?
    return @@buf.shift
  end
  c = nil
  loop do
    result = @@input.wait_readable(0.1)
    next if result.nil?
    c = @@input.read(1)
    break
  end
  c&.ord
end

def self.reset(encoding: nil)
  @@pasting = false
  if encoding
    @@encoding = encoding
  elsif defined?(@@encoding)
    remove_class_variable(:@@encoding)
  end
end
