formatDuration

Formats a duration in human-readable format

Description

Return human-readable duration string i.e. "9 months 2 days"

Usage

// CommonJS
var formatDuration = require('date-fns/formatDuration')
// ES 2015
import formatDuration from 'date-fns/formatDuration'
// ESM
import { formatDuration } from 'date-fns'

Syntax

formatDuration(duration, [options])

Arguments

Name Description
duration

the duration to format
options

an object with options.
options.format

the array of units to format
options.zero

should zeros be included in the output?
options.delimiter

delimiter string
options.locale

the locale object. See [Locale]{@link https://date-fns.org/docs/Locale}

Returns

Description

the formatted date string

Exceptions

Type Description
TypeError

1 argument required

Examples

// Format full duration
formatDuration({
  years: 2,
  months: 9,
  weeks: 1,
  days: 7,
  hours: 5,
  minutes: 9,
  seconds: 30
})
//=> '2 years 9 months 1 week 7 days 5 hours 9 minutes 30 seconds'
// Format partial duration
formatDuration({ months: 9, days: 2 })
//=> '9 months 2 days'
// Customize the format
formatDuration(
  {
    years: 2,
    months: 9,
    weeks: 1,
    days: 7,
    hours: 5,
    minutes: 9,
    seconds: 30
  },
  { format: ['months', 'weeks'] }
) === '9 months 1 week'
// Customize the zeros presence
formatDuration({ years: 0, months: 9 })
//=> '9 months'
formatDuration({ years: 0, months: 9 }, { zero: true })
//=> '0 years 9 months'
// Customize the delimiter
formatDuration({ years: 2, months: 9, weeks: 3 }, { delimiter: ', ' })
//=> '2 years, 9 months, 3 weeks'
