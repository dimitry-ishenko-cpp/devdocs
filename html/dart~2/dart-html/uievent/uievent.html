factory UIEvent(String type,
    {Window? view,
    int detail: 0,
    bool canBubble: true,
    bool cancelable: true}) {
  if (view == null) {
    view = window;
  }
  UIEvent e = document._createEvent("UIEvent") as UIEvent;
  e._initUIEvent(type, canBubble, cancelable, view, detail);
  return e;
}
