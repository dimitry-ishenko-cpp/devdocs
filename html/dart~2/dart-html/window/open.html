WindowBase open(String url, String name, [String? options]) {
  if (options == null) {
    return _DOMWindowCrossFrame._createSafe(_open2(url, name));
  } else {
    return _DOMWindowCrossFrame._createSafe(_open3(url, name, options));
  }
}
