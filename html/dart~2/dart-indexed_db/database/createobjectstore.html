ObjectStore createObjectStore(String name, {keyPath, bool? autoIncrement}) {
  var options = {};
  if (keyPath != null) {
    options['keyPath'] = keyPath;
  }
  if (autoIncrement != null) {
    options['autoIncrement'] = autoIncrement;
  }

  return _createObjectStore(name, options);
}
