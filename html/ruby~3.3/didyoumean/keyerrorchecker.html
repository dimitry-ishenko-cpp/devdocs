def corrections
  @corrections ||= exact_matches.empty? ? SpellChecker.new(dictionary: @keys).correct(@key).map(&:inspect) : exact_matches
end

def exact_matches
  @exact_matches ||= @keys.select { |word| @key == word.to_s }.map(&:inspect)
end
