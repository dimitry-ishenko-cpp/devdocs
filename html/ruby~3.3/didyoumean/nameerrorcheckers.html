def new(exception)
  case exception.original_message
  when /uninitialized constant/
    ClassNameChecker
  when /undefined local variable or method/,
       /undefined method/,
       /uninitialized class variable/,
       /no member '.*' in struct/
    VariableNameChecker
  else
    NullChecker
  end.new(exception)
end
