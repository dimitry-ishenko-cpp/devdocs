def sym(symbol)
  @handlers.each{|handle|
    if( handle )
      begin
        addr = handle.sym(symbol)
        return addr
      rescue DLError
      end
    end
  }
  return nil
end

Returns the address as an Integer from any handlers with the function named symbol.

Raises a DLError if the handle is closed.
