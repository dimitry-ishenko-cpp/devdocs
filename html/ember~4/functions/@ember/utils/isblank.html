A value is blank if it is empty or a whitespace string.

import { isBlank } from '@ember/utils';

isBlank(null);            // true
isBlank(undefined);       // true
isBlank('');              // true
isBlank([]);              // true
isBlank('\n\t');          // true
isBlank('  ');            // true
isBlank({});              // false
isBlank('\n\t Hello');    // false
isBlank('Hello world');   // false
isBlank([1,2,3]);         // false
