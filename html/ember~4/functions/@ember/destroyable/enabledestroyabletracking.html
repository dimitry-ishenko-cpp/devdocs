import { enableDestroyableTracking } from '@ember/destroyable';

This function instructs the destroyable system to keep track of all destroyables (their children, destructors, etc). This enables a future usage of assertDestroyablesDestroyed to be used to ensure that all destroyable tasks (registered destructors and associated children) have completed when assertDestroyablesDestroyed is called.
