Description

Convenience definitions for working with ByteStrings.

type Parser = Parsec ByteString () Source

type GenParser t st = Parsec ByteString st Source

parseFromFile :: Parser a -> FilePath -> IO (Either ParseError a) Source

parseFromFile p filePath runs a strict bytestring parser p on the input read from filePath using readFile. Returns either a ParseError (Left) or a value of type a (Right).

 main    = do{ result <- parseFromFile numbers "digits.txt"
             ; case result of
                 Left err  -> print err
                 Right xs  -> print (sum xs)
             }
