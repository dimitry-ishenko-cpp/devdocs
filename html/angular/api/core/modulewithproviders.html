ModuleWithProviders

interface

A wrapper around an NgModule that associates it with providers. Usage without a generic type is deprecated.

interface ModuleWithProviders<T> {
  ngModule: Type<T>
  providers?: Array<Provider | EnvironmentProviders>
}
