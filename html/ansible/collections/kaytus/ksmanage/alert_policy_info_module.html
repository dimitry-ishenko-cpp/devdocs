Examples

- name: Alert test
  hosts: ksmanage
  connection: local
  gather_facts: false
  vars:
    ksmanage:
      host: "{{ ansible_ssh_host }}"
      username: "{{ username }}"
      password: "{{ password }}"

  tasks:

  - name: "Get alert policy"
    kaytus.ksmanage.alert_policy_info:
      provider: "{{ ksmanage }}"
