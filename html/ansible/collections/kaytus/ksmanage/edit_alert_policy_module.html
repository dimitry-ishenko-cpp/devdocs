Synopsis

  • Set alert policy on kaytus Server.

Requirements

The below requirements are needed on the host that executes this module.

  • Python 3.7+
  • inManage

Examples

- name: Alert policy test
  hosts: ksmanage
  connection: local
  gather_facts: false
  vars:
    ksmanage:
      host: "{{ ansible_ssh_host }}"
      username: "{{ username }}"
      password: "{{ password }}"

  tasks:

  - name: "Set alert policy"
    kaytus.ksmanage.edit_alert_policy:
      id: 1
      status: "enable"
      type: "snmp"
      destination: "100.2.2.2"
      channel: "shared"
      provider: "{{ ksmanage }}"

  - name: "Set alert policy"
    kaytus.ksmanage.edit_alert_policy:
      id: 1
      status: "disable"
      provider: "{{ ksmanage }}"
