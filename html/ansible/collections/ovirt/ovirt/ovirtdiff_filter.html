Examples

- name: Get VM myvm
  ovirt_vm_info:
    auth: "{{ ovirt_auth }}"
    pattern: 'name={{ myvm }}'
    next_run: false
  register: vm

- name: Get next_run of VM myvm
  ovirt_vm_info:
    auth: "{{ ovirt_auth }}"
    pattern: 'name={{ myvm }}'
    next_run: true
  register: vm_next_run

- name: Print what will be changed in next run of the VM
  debug:
    msg: "{{ vm.ovirt_vms[0] | ovirt.ovirt.ovirtdiff(vm_next_run.ovirt_vms[0]) }}"
