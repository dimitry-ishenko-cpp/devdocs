Examples

- name: Ipv6 test
  hosts: ism
  connection: local
  gather_facts: no
  vars:
    ism:
      host: "{{ ansible_ssh_host }}"
      username: "{{ username }}"
      password: "{{ password }}"

  tasks:

  - name: "Set ipv6 information"
    inspur.sm.edit_ipv6:
      interface_name: "eth0"
      ipv6_status: "disable"
      provider: "{{ ism }}"

  - name: "Set ipv6 information"
    inspur.sm.edit_ipv6:
      interface_name: "eth0"
      ipv6_status: "enable"
      ipv6_dhcp_enable: "dhcp"
      provider: "{{ ism }}"

  - name: "Set ipv6 information"
    inspur.sm.edit_ipv6:
      interface_name: "eth0"
      ipv6_status: "enable"
      ipv6_dhcp_enable: "static"
      ipv6_address: "::ffff:100:2:36:10"
      ipv6_index: 12
      ipv6_prefix: 16
      ipv6_gateway: "::"
      provider: "{{ ism }}"
