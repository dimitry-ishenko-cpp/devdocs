Examples

- name: Set user group test
  hosts: ism
  connection: local
  gather_facts: no
  vars:
    ism:
      host: "{{ ansible_ssh_host }}"
      username: "{{ username }}"
      password: "{{ password }}"

  tasks:

  - name: "Set user group"
    inspur.sm.edit_user_group:
      name: "test"
      pri: "administrator"
      provider: "{{ ism }}"
