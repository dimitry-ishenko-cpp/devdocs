Synopsis

  • Only the M6 models support this feature.

Examples

- name: Smtp com test
  hosts: ism
  no_log: true
  connection: local
  gather_facts: no
  vars:
    ism:
      host: "{{ ansible_ssh_host }}"
      username: "{{ username }}"
      password: "{{ password }}"

  tasks:

  - name: "Set smtp com information"
    inspur.sm.edit_smtp_com:
      status: "disable"
      provider: "{{ ism }}"

  - name: "Set smtp com information"
    inspur.sm.edit_smtp_com:
      status: "enable"
      server_ip: "100.2.2.2"
      email: "inspur@Inspur.com"
      server_auth: "enable"
      server_username: "admin"
      server_password: "1234qwer!@#$"
      provider: "{{ ism }}"
