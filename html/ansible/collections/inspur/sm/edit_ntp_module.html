Examples

- name: NTP test
  hosts: ism
  connection: local
  gather_facts: no
  vars:
    ism:
      host: "{{ ansible_ssh_host }}"
      username: "{{ username }}"
      password: "{{ password }}"

  tasks:

  - name: "Set ntp"
    inspur.sm.edit_ntp:
      auto_date: "enable"
      server2: "time.nist.gov"
      provider: "{{ ism }}"

  - name: "Set ntp"
    inspur.sm.edit_ntp:
      auto_date: "disable"
      ntp_time: "20200609083600"
      provider: "{{ ism }}"

  - name: "set ntp"
    inspur.sm.edit_ntp:
      time_zone: 8
      provider: "{{ ism }}"
