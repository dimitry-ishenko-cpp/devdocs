Examples

- name: Event log policy test
  hosts: ism
  connection: local
  gather_facts: no
  vars:
    ism:
      host: "{{ ansible_ssh_host }}"
      username: "{{ username }}"
      password: "{{ password }}"

  tasks:

  - name: "Get event log policy information"
    inspur.sm.event_log_policy_info:
      provider: "{{ ism }}"
