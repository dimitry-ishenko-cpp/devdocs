fun <T> mutableListOf(): MutableList<T>

Returns an empty new MutableList.

import kotlin.test.*

fun main(args: Array<String>) {
//sampleStart
val list = mutableListOf<Int>()
println("list.isEmpty() is ${list.isEmpty()}") // true

list.addAll(listOf(1, 2, 3))
println(list) // [1, 2, 3]
//sampleEnd
}
