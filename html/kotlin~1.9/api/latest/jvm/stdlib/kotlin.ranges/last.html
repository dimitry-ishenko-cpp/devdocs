last

kotlin-stdlib / kotlin.ranges / last
Platform and version requirements: JVM (1.7), JS (1.7), Native (1.7) 
fun IntProgression.last(): Int
fun LongProgression.last(): Long
fun CharProgression.last(): Char
fun UIntProgression.last(): UInt
fun ULongProgression.last(): ULong

Returns the last element.

import kotlin.test.*

fun main(args: Array<String>) {
//sampleStart
val list = listOf(1, 2, 3, 4)
println(list.last()) // 4
println(list.last { it % 2 == 1 }) // 3
println(list.lastOrNull { it < 0 }) // null
// list.last { it < 0 } //  will fail

val emptyList = emptyList<Int>()
println(emptyList.lastOrNull()) // null
// emptyList.last() //  will fail
//sampleEnd
}

Exceptions

NoSuchElementException - if the progression is empty.
