factory CanvasElement({int? width, int? height}) {
  CanvasElement e = JS<CanvasElement>(
      'returns:CanvasElement;creates:CanvasElement;new:true',
      '#.createElement(#)',
      document,
      "canvas");
  if (width != null) e.width = width;
  if (height != null) e.height = height;
  return e;
}
