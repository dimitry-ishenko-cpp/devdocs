void clear() {
  _modificationCount++;
  if (isEmpty) return;

  E next = _first!;
  do {
    E entry = next;
    next = entry._next!;
    entry._next = entry._previous = entry._list = null;
  } while (!identical(next, _first));

  _first = null;
  _length = 0;
}
