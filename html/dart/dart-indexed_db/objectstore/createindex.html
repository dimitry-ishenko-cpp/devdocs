Index createIndex(String name, keyPath, {bool? unique, bool? multiEntry}) {
  var options = {};
  if (unique != null) {
    options['unique'] = unique;
  }
  if (multiEntry != null) {
    options['multiEntry'] = multiEntry;
  }

  return _createIndex(name, keyPath, options);
}
