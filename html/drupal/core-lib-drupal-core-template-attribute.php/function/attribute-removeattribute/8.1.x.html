public function removeAttribute() {
  $args = func_get_args();
  foreach ($args as $arg) {
    // Support arrays or multiple arguments.
    if (is_array($arg)) {
      foreach ($arg as $value) {
        unset($this->storage[$value]);
      }
    }
    else {
      unset($this->storage[$arg]);
    }
  }

  return $this;
}
