protected function escape(array $parameters) {
  $args = array();

  foreach ($parameters as $key => $value) {
    if (is_array($value)) {
      $args[$key] = $this->escape($value);
    }
    elseif (is_string($value)) {
      $args[$key] = str_replace('%', '%%', $value);
    }
    else {
      $args[$key] = $value;
    }
  }

  return $args;
}
