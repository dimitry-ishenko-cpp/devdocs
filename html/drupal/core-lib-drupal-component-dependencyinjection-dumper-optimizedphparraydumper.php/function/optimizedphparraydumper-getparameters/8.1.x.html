protected function getParameters() {
  if (!$this->container->getParameterBag()->all()) {
    return array();
  }

  $parameters = $this->container->getParameterBag()->all();
  $is_frozen = $this->container->isFrozen();
  return $this->prepareParameters($parameters, $is_frozen);
}
