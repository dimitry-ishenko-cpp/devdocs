public function set($key, $value) {
  $value = $this->castSafeStrings($value);
  // The dot/period is a reserved character; it may appear between keys, but
  // not within keys.
  if (is_array($value)) {
    $this->validateKeys($value);
  }
  $parts = explode('.', $key);
  if (count($parts) == 1) {
    $this->data[$key] = $value;
  }
  else {
    NestedArray::setValue($this->data, $parts, $value);
  }
  return $this;
}
