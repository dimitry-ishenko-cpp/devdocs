protected function validateKeys(array $data) {
  foreach ($data as $key => $value) {
    if (strpos($key, '.') !== FALSE) {
      throw new ConfigValueException("$key key contains a dot which is not supported.");
    }
    if (is_array($value)) {
      $this->validateKeys($value);
    }
  }
}
