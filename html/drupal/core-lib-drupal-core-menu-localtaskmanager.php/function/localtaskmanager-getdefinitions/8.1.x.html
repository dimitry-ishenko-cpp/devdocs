public function getDefinitions() {
  $definitions = parent::getDefinitions();

  $count = 0;
  foreach ($definitions as &$definition) {
    if (isset($definition['weight'])) {
      // Add some micro weight.
      $definition['weight'] += ($count++) * 1e-6;
    }
  }

  return $definitions;
}
