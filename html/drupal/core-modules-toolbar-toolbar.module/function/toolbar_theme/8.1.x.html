function toolbar_theme($existing, $type, $theme, $path) {
  $items['toolbar'] = array(
    'render element' => 'element',
  );
  $items['menu__toolbar'] = array(
    'base hook' => 'menu',
    'variables' => array('items' => array(), 'attributes' => array()),
  );

  return $items;
}
