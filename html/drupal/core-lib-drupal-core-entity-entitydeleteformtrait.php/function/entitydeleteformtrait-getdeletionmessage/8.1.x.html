protected function getDeletionMessage() {
  $entity = $this->getEntity();
  return $this->t('The @entity-type %label has been deleted.', array(
    '@entity-type' => $entity->getEntityType()->getLowercaseLabel(),
    '%label' => $entity->label(),
  ));
}
