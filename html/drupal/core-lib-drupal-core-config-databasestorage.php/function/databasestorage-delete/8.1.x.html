public function delete($name) {
  $options = array('return' => Database::RETURN_AFFECTED) + $this->options;
  return (bool) $this->connection->delete($this->table, $options)
    ->condition('collection', $this->collection)
    ->condition('name', $name)
    ->execute();
}
