public function orig() {
  $orig = new HWLDFWordAccumulator();

  foreach ($this->edits as $edit) {
    if ($edit->type == 'copy') {
      $orig->addWords($edit->orig);
    }
    elseif ($edit->orig) {
      $orig->addWords($edit->orig, 'mark');
    }
  }
  $lines = $orig->getLines();
  return $lines;
}
