function system_theme_suggestions_html(array $variables) {
  if (\Drupal::service('path.matcher')->isFrontPage()) {
    $path_args = [''];
  }
  else {
    $path_args = explode('/', ltrim(\Drupal::service('path.current')->getPath(), '/'));
  }
  return theme_get_suggestions($path_args, 'html');
}
