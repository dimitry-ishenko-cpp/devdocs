public function getHookInfo() {
  if (!isset($this->hookInfo)) {
    if ($cache = $this->cacheBackend->get('hook_info')) {
      $this->hookInfo = $cache->data;
    }
    else {
      $this->buildHookInfo();
      $this->cacheBackend->set('hook_info', $this->hookInfo);
    }
  }
  return $this->hookInfo;
}
