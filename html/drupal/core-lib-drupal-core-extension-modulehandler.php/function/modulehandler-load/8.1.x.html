public function load($name) {
  if (isset($this->loadedFiles[$name])) {
    return TRUE;
  }

  if (isset($this->moduleList[$name])) {
    $this->moduleList[$name]->load();
    $this->loadedFiles[$name] = TRUE;
    return TRUE;
  }
  return FALSE;
}
