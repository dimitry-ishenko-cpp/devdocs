protected function handleException(\Exception $e, $request, $type) {
  if ($e instanceof HttpExceptionInterface) {
    $response = new Response($e->getMessage(), $e->getStatusCode());
    $response->headers->add($e->getHeaders());
    return $response;
  }
  else {
    throw $e;
  }
}
