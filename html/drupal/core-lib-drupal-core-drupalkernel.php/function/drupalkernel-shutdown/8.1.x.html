public function shutdown() {
  if (FALSE === $this->booted) {
    return;
  }
  $this->container->get('stream_wrapper_manager')->unregister();
  $this->booted = FALSE;
  $this->container = NULL;
  $this->moduleList = NULL;
  $this->moduleData = array();
}
