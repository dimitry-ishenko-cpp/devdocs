protected function getModulesParameter() {
  $extensions = array();
  foreach ($this->moduleList as $name => $weight) {
    if ($data = $this->moduleData($name)) {
      $extensions[$name] = array(
        'type' => $data->getType(),
        'pathname' => $data->getPathname(),
        'filename' => $data->getExtensionFilename(),
      );
    }
  }
  return $extensions;
}
