public function releaseAll($lock_id = NULL) {
  // Only attempt to release locks if any were acquired.
  if (!empty($this->locks)) {
    $this->locks = array();
    if (empty($lock_id)) {
      $lock_id = $this->getLockId();
    }
    $this->database->delete('semaphore')
      ->condition('value', $lock_id)
      ->execute();
  }
}
