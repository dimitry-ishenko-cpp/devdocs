protected function _flushLine($new_tag) {
  $this->_flushGroup($new_tag);
  if ($this->line != '') {
    array_push($this->lines, $this->line);
  }
  else {
    // make empty lines visible by inserting an NBSP
    array_push($this->lines, $this::NBSP);
  }
  $this->line = '';
}
