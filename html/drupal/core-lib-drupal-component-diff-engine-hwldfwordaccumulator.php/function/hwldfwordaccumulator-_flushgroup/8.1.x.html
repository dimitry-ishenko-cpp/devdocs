protected function _flushGroup($new_tag) {
  if ($this->group !== '') {
    if ($this->tag == 'mark') {
      $this->line = $this->line . '<span class="diffchange">' . $this->group . '</span>';
    }
    else {
      $this->line = $this->line . $this->group;
    }
  }
  $this->group = '';
  $this->tag = $new_tag;
}
