function setWithExpire($key, $value, $expire) {
  $this->connection->merge($this->table)
    ->keys(array(
      'name' => $key,
      'collection' => $this->collection,
    ))
    ->fields(array(
      'value' => $this->serializer->encode($value),
      'expire' => REQUEST_TIME + $expire,
    ))
    ->execute();
}
