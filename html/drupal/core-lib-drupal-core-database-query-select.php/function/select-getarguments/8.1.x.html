public function getArguments(PlaceholderInterface $queryPlaceholder = NULL) {
  if (!isset($queryPlaceholder)) {
    $queryPlaceholder = $this;
  }
  $this->compile($this->connection, $queryPlaceholder);
  return $this->arguments();
}
