protected function saveEntitySchemaData(EntityTypeInterface $entity_type, $schema) {
  $data = $this->getEntitySchemaData($entity_type, $schema);
  $this->installedStorageSchema()->set($entity_type->id() . '.entity_schema_data', $data);
}
