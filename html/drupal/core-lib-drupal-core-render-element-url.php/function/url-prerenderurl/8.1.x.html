public static function preRenderUrl($element) {
  $element['#attributes']['type'] = 'url';
  Element::setAttributes($element, array('id', 'name', 'value', 'size', 'maxlength', 'placeholder'));
  static::setAttributes($element, array('form-url'));

  return $element;
}
