public function getDefinitions() {
  $definitions = array();
  foreach ($this->schemaStorage->readMultiple($this->schemaStorage->listAll()) as $schema) {
    foreach ($schema as $type => $definition) {
      $definitions[$type] = $definition;
    }
  }
  return $definitions;
}
