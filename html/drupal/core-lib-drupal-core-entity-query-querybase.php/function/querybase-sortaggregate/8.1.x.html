public function sortAggregate($field, $function, $direction = 'ASC', $langcode = NULL) {
  $alias = $this->getAggregationAlias($field, $function);

  $this->sortAggregate[$alias] = array(
    'field' => $field,
    'function' => $function,
    'direction' => $direction,
    'langcode' => $langcode,
  );
  $this->aggregate($field, $function, $langcode, $alias);

  return $this;
}
