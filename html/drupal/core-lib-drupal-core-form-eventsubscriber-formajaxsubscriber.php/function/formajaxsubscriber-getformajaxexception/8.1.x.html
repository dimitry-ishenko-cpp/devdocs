protected function getFormAjaxException(\Exception $e) {
  $exception = NULL;
  while ($e) {
    if ($e instanceof FormAjaxException) {
      $exception = $e;
      break;
    }

    $e = $e->getPrevious();
  }
  return $exception;
}
