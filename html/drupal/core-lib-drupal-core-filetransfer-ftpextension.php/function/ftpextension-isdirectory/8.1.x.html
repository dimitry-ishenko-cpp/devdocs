public function isDirectory($path) {
  $result = FALSE;
  $curr = ftp_pwd($this->connection);
  if (@ftp_chdir($this->connection, $path)) {
    $result = TRUE;
  }
  ftp_chdir($this->connection, $curr);
  return $result;
}
