public function fetchColumn($index = 0) {
  if (isset($this->currentRow) && isset($this->columnNames[$index])) {
    // We grab the value directly from $this->data, and format it.
    $return = $this->currentRow[$this->columnNames[$index]];
    $this->next();
    return $return;
  }
  else {
    return FALSE;
  }
}
