public function delete($key) {
  $this->lazyLoadCache();
  unset($this->storage[$key]);
  $this->keysToRemove[$key] = $key;
  // The key might have been marked for persisting.
  unset($this->keysToPersist[$key]);
  $this->invalidateCache();
}
