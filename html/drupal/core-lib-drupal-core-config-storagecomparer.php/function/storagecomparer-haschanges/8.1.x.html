public function hasChanges() {
  foreach ($this->getAllCollectionNames() as $collection) {
    foreach (array('delete', 'create', 'update', 'rename') as $op) {
      if (!empty($this->changelist[$collection][$op])) {
        return TRUE;
      }
    }
  }
  return FALSE;
}
