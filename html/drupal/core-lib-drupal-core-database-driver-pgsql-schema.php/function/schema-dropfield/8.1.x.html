public function dropField($table, $field) {
  if (!$this->fieldExists($table, $field)) {
    return FALSE;
  }

  $this->connection->query('ALTER TABLE {' . $table . '} DROP COLUMN "' . $field . '"');
  $this->resetTableInformation($table);
  return TRUE;
}
