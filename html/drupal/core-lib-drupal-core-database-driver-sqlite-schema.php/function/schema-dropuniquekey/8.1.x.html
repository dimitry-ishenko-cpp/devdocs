public function dropUniqueKey($table, $name) {
  if (!$this->indexExists($table, $name)) {
    return FALSE;
  }

  $info = $this->getPrefixInfo($table);

  $this->connection->query('DROP INDEX ' . $info['schema'] . '.' . $info['table'] . '_' . $name);
  return TRUE;
}
