public static function buildProxyNamespace($class_name) {
  $proxy_classname = static::buildProxyClassName($class_name);

  preg_match('/(.+)\\\\[a-zA-Z0-9]+/', $proxy_classname, $match);
  $proxy_namespace = $match[1];
  return $proxy_namespace;
}
