public static function buildProxyClassName($class_name) {
  $match = [];
  preg_match('/([a-zA-Z0-9_]+\\\\[a-zA-Z0-9_]+)\\\\(.+)/', $class_name, $match);
  $root_namespace = $match[1];
  $rest_fqcn = $match[2];
  $proxy_class_name = $root_namespace . '\\ProxyClass\\' . $rest_fqcn;

  return $proxy_class_name;
}
