public function getPath($request = NULL) {
  if (!isset($request)) {
    $request = $this->requestStack->getCurrentRequest();
  }
  if (!isset($this->paths[$request])) {
    $this->paths[$request] = $request->getPathInfo();
  }

  return $this->paths[$request];
}
