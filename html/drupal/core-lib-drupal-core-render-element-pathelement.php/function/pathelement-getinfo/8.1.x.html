public function getInfo() {
  $info = parent::getInfo();
  $class = get_class($this);
  $info['#validate_path'] = TRUE;
  $info['#convert_path'] = self::CONVERT_ROUTE;
  $info['#element_validate'] = array(
    array($class, 'validateMatchedPath'),
  );
  return $info;
}
