protected static function sortGraphByWeight(array $a, array $b) {
  $weight_cmp = SortArray::sortByKeyInt($a, $b, 'weight');

  if ($weight_cmp === 0) {
    return SortArray::sortByKeyString($a, $b, 'name');
  }
  return $weight_cmp;
}
