protected function updateFieldLangcodes($langcode) {
  foreach ($this->fields as $name => $items) {
    if (!empty($items[LanguageInterface::LANGCODE_DEFAULT])) {
      $items[LanguageInterface::LANGCODE_DEFAULT]->setLangcode($langcode);
    }
  }
}
