public function setCustomStorage($custom_storage) {
  if (!$custom_storage && $this->isComputed()) {
    throw new \LogicException("Entity storage cannot store a computed field.");
  }
  $this->definition['custom_storage'] = $custom_storage;
  return $this;
}
