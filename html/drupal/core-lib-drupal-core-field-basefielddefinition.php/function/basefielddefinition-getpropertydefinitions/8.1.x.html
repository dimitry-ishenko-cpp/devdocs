public function getPropertyDefinitions() {
  if (!isset($this->propertyDefinitions)) {
    $class = $this->getFieldItemClass();
    $this->propertyDefinitions = $class::propertyDefinitions($this);
  }
  return $this->propertyDefinitions;
}
