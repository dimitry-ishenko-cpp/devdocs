public function getPropertyDefinition($name) {
  if (!isset($this->propertyDefinitions)) {
    $this->getPropertyDefinitions();
  }
  if (isset($this->propertyDefinitions[$name])) {
    return $this->propertyDefinitions[$name];
  }
}
