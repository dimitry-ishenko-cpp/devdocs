public function getInfo() {
  $class = get_class($this);
  return array(
    '#input' => TRUE,
    '#js_select' => TRUE,
    '#multiple' => TRUE,
    '#responsive' => TRUE,
    '#sticky' => FALSE,
    '#pre_render' => array(
      array($class, 'preRenderTable'),
      array($class, 'preRenderTableselect'),
    ),
    '#process' => array(
      array($class, 'processTableselect'),
    ),
    '#options' => array(),
    '#empty' => '',
    '#theme' => 'table__tableselect',
  );
}
