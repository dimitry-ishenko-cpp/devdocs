public function isFile($path) {
  $file = escapeshellarg($path);
  $cmd = "[ -f {$file} ] && echo 'yes'";
  if ($output = @ssh2_exec($this->connection, $cmd)) {
    if ($output == 'yes') {
      return TRUE;
    }
    return FALSE;
  }
  else {
    throw new FileTransferException('Cannot check @path.', NULL, array('@path' => $path));
  }
}
