protected function getParentDepthLimit($id) {
  if ($id) {
    $limit = $this->menuLinkTree->maxDepth() - $this->menuLinkTree->getSubtreeHeight($id);
  }
  else {
    $limit = $this->menuLinkTree->maxDepth() - 1;
  }
  return $limit;
}
