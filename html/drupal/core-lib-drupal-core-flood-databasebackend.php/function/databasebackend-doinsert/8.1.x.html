protected function doInsert($name, $window, $identifier) {
  $this->connection->insert(static::TABLE_NAME)
    ->fields(array(
      'event' => $name,
      'identifier' => $identifier,
      'timestamp' => REQUEST_TIME,
      'expiration' => REQUEST_TIME + $window,
    ))
    ->execute();
}
