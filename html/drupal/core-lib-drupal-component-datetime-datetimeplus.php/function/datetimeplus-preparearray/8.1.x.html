public static function prepareArray($array, $force_valid_date = FALSE) {
  if ($force_valid_date) {
    $now = new \DateTime();
    $array += array(
      'year' => $now->format('Y'),
      'month' => 1,
      'day' => 1,
      'hour' => 0,
      'minute' => 0,
      'second' => 0,
    );
  }
  else {
    $array += array(
      'year' => '',
      'month' => '',
      'day' => '',
      'hour' => '',
      'minute' => '',
      'second' => '',
    );
  }
  return $array;
}
