public function setFormState(array $form_state_additions) {
  foreach ($form_state_additions as $key => $value) {
    if (property_exists($this, $key)) {
      $this->{$key} = $value;
    }
    else {
      $this->set($key, $value);
    }
  }
  return $this;
}
