function _filter_url_trim($text, $length = NULL) {
  static $_length;
  if ($length !== NULL) {
    $_length = $length;
  }

  if (isset($_length)) {
    $text = Unicode::truncate($text, $_length, FALSE, TRUE);
  }

  return $text;
}
