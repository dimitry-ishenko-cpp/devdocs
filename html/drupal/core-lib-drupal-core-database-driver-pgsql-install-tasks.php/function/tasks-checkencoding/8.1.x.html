protected function checkEncoding() {
  try {
    if (db_query('SHOW server_encoding')->fetchField() == 'UTF8') {
      $this->pass(t('Database is encoded in UTF-8'));
    }
    else {
      $this->fail(t('The %driver database must use %encoding encoding to work with Drupal. Recreate the database with %encoding encoding. See <a href="INSTALL.pgsql.txt.html">INSTALL.pgsql.txt</a> for more details.', array(
        '%encoding' => 'UTF8',
        '%driver' => $this->name(),
      )));
    }
  }
  catch (\Exception $e) {
    $this->fail(t('Drupal could not determine the encoding of the database was set to UTF-8'));
  }
}
