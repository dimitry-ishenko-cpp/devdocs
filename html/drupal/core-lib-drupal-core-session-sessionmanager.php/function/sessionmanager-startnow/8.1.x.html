protected function startNow() {
  if ($this->isCli()) {
    return FALSE;
  }

  if ($this->startedLazy) {
    // Save current session data before starting it, as PHP will destroy it.
    $session_data = $_SESSION;
  }

  $result = parent::start();

  // Restore session data.
  if ($this->startedLazy) {
    $_SESSION = $session_data;
    $this->loadSession();
  }

  return $result;
}
