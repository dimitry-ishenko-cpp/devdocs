public function delete($uid) {
  // Nothing to do if we are not allowed to change the session.
  if (!$this->writeSafeHandler->isSessionWritable() || $this->isCli()) {
    return;
  }
  $this->connection->delete('sessions')
    ->condition('uid', $uid)
    ->execute();
}
