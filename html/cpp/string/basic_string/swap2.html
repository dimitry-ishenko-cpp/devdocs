std::swap(std::basic_string)

Defined in header <string>

           std::basic_string<CharT, Traits, Alloc>& rhs );

           std::basic_string<CharT, Traits, Alloc>& rhs ) noexcept(/* see below */);
 (since C++17)

constexpr void
    swap( std::basic_string<CharT, Traits, Alloc>& lhs,

 (since C++20)

Specializes the std::swap algorithm for std::basic_string. Swaps the contents of lhs and rhs.

Equivalent to lhs.swap(rhs).

Parameters

lhs, rhs - strings whose contents to swap

Return value

(none)

Complexity

Constant.

Exceptions

noexcept specification:
noexcept(noexcept(lhs.swap(rhs)))
 (since C++17)

Example

#include <iostream>
#include <string>
 
int main()
{
    std::string a = "AAA";
    std::string b = "BBBB";
 
    std::cout << "Before swap:\n"
                 "a = " << a << "\n"
                 "b = " << b << "\n\n";
 
    std::swap(a, b);
 
    std::cout << "After swap:\n"
                 "a = " << a << "\n"
                 "b = " << b << '\n';
}

Output:

Before swap:
a = AAA
b = BBBB
 
After swap:
a = BBBB
b = AAA
