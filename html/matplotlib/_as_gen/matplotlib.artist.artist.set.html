Description

scalar or None

animated

bool

Bbox

bool

str

bool

object

bool

path_effects

AbstractPathEffect

picker

None or bool or float or callable

rasterized

bool

sketch_params

snap

bool or None

transform

Transform

url

str

visible

bool

float
